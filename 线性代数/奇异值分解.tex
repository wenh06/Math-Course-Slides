%------------------------------------------------

\begin{frame}

\begin{block}{еј•зҗҶ}
\centering
$MM^T$дёҺ$M^TM$зҡ„зү№еҫҒеҖјйқһиҙҹпјҢдё”дәҢиҖ…зҡ„жӯЈзү№еҫҒеҖјд№ӢйӣҶзӣёеҗҢгҖӮ
\end{block}

\pause

\begin{proof}[иҜҒжҳҺ\nopunct]
д»»еҸ–$MM^T$зҡ„дёҖдёӘзү№еҫҒеҖј$\lambda$д»ҘеҸҠзӣёеә”зҡ„дёҖдёӘзү№еҫҒеҗ‘йҮҸ$\mathbf{x}$пјҢеҚіжңү$MM^T\mathbf{x} = \lambda\mathbf{x}$гҖӮйӮЈд№Ҳ$\mathbf{x}^TMM^T\mathbf{x} = \mathbf{x}^T\lambda\mathbf{x}$пјҢеҚі
\begin{equation}
\label{eq1}
\Vert M^T\mathbf{x} \Vert^2 = \lambda \Vert \mathbf{x} \Vert^2.
\end{equation}
жүҖд»Ҙ$\lambda$йқһиҙҹгҖӮеҗҢзҗҶ$M^TM$зҡ„зү№еҫҒеҖјд№ҹйғҪжҳҜйқһиҙҹе®һж•°гҖӮ\newline\pause
и®ҫ$\lambda > 0$,иҝҳжҳҜз”ұ$MM^T\mathbf{x} = \lambda\mathbf{x}$пјҢжңү
\begin{equation}
\label{eq2}
M^TMM^T\mathbf{x} = \lambda M^T\mathbf{x},
\end{equation}
еҸҲз”ұ\eqref{eq1}ејҸзҹҘ$M^T\mathbf{x}$йқһйӣ¶еҗ‘йҮҸпјҢж•…$\lambda$д№ҹжҳҜ$M^TM$зҡ„жӯЈзү№еҫҒеҖјгҖӮ
\end{proof}

\end{frame}

%------------------------------------------------

\begin{frame}

\begin{block}{еҘҮејӮеҖјеҲҶи§Је®ҡзҗҶзҡ„иҜҒжҳҺ}
жҠҠ$MM^T$дёҺ$M^TM$зҡ„жӯЈзү№еҫҒеҖји®°дёә$\sigma_1^2 \geqslant \cdots \geqslant \sigma_r^2 > 0,$ е…¶дёӯ$\sigma_i>0.$ и®ҫ$M^TM$йҖҡиҝҮ$n$йҳ¶жӯЈдәӨйҳө$V = (\mathbf{v}_1,\ldots,\mathbf{v}_n)$еҜ№и§’еҢ–пјҡ
$$M^TM = V \begin{pmatrix}
\sigma_1^2 & & & \\ & \ddots & & \\ & & \sigma_r^2 & \\ & & & \bigzero \end{pmatrix}_n\!\! V^T.$$
жңү$M^TM\mathbf{v}_i = \sigma_i^2\mathbf{v}_i,$ д»Ө$\mathbf{u}_i = \dfrac{M\mathbf{v}_i}{\sigma_i},$ йӮЈд№ҲеҸҜд»ҘиҜҒжҳҺпјҡ

\pause

\begin{enumerate}
\item $\mathbf{u}_1, \ldots, \mathbf{u}_r$жҳҜ$m$йҳ¶е®һеҜ№з§°йҳө$MM^T$зҡ„еҚ•дҪҚжӯЈдәӨзҡ„зү№еҫҒеҗ‘йҮҸпјӣ
\item $M\mathbf{v}_i = \sigma_i\mathbf{u}_i, \ M^T\mathbf{u}_i = \sigma_i\mathbf{v}_i.$
\end{enumerate}
\end{block}

\end{frame}

\begin{frame}

\begin{block}{еҘҮејӮеҖјеҲҶи§Је®ҡзҗҶзҡ„иҜҒжҳҺ}

\begin{frame}

\begin{block}{еҘҮејӮеҖјеҲҶи§Јзҡ„дёҖз§Қжӣҙзҙ§еҮ‘зҡ„еҪўејҸ}

\section{дҫӢеӯҗдёҺеә”з”Ё}

%------------------------------------------------

\begin{frame}

\begin{block}{еҘҮејӮеҖјеҲҶи§Јзҡ„дҫӢеӯҗ}
и®Ўз®—зҹ©йҳө$M = \begin{pmatrix} \cos\theta & -\sin\theta \\ \sin\theta & \cos\theta \end{pmatrix}$зҡ„еҘҮејӮеҖјеҲҶи§ЈгҖӮ

\vspace{1em}
\pause

и§Јпјҡжңү
$$M^TM = \begin{pmatrix} \cos\theta & \sin\theta \\ -\sin\theta & \cos\theta \end{pmatrix} \begin{pmatrix} \cos\theta & -\sin\theta \\ \sin\theta & \cos\theta \end{pmatrix} = \begin{pmatrix} 1 & 0 \\ 0 & 1 \end{pmatrix}.$$
дәҺжҳҜжңүеҘҮејӮеҖј$\sigma_1 = \sigma_2 = 1$. жҲ‘д»¬еҸҜд»ҘйҖүеҸ–
$$\mathbf{v}_1 = \begin{pmatrix} 1 \\ 0 \end{pmatrix}, \quad \mathbf{v}_2 = \begin{pmatrix} 0 \\ 1 \end{pmatrix}$$
д»ҺиҖҢжңү
$$\mathbf{u}_1 = M\mathbf{v}_1 = \begin{pmatrix} \cos\theta \\ \sin\theta \end{pmatrix}, \quad \mathbf{u}_2 = M\mathbf{v}_2 = \begin{pmatrix} -\sin\theta \\ \cos\theta \end{pmatrix}$$
\end{block}

\end{frame}

%------------------------------------------------

\begin{frame}

\begin{block}{еҘҮејӮеҖјеҲҶи§Јзҡ„дҫӢеӯҗпјҲз»ӯпјү}
дәҺжҳҜ$M = \begin{pmatrix} \cos\theta & -\sin\theta \\ \sin\theta & \cos\theta \end{pmatrix}$зҡ„еҘҮејӮеҖјеҲҶи§ЈеҸҜд»ҘжҳҜ
$$M = U\Sigma V^T = \begin{pmatrix} \cos\theta & -\sin\theta \\ \sin\theta & \cos\theta \end{pmatrix} \cdot I_2 \cdot I_2.$$

\vspace{1em}
\pause

жҲ‘д»¬д№ҹеҸҜд»ҘеҸҰйҖү
$$\mathbf{v}_1 = \begin{pmatrix} \cos\alpha \\ \sin\alpha \end{pmatrix}, \quad \mathbf{v}_2 = \begin{pmatrix} -\sin\alpha \\ \cos\alpha \end{pmatrix},$$
дәҺжҳҜпјҢзӣёеә”зҡ„еҘҮејӮеҖјеҲҶи§Јдёә
$$M = U\Sigma V^T = \begin{pmatrix} \cos(\theta+\alpha) & -\sin(\theta+\alpha) \\ \sin(\theta+\alpha) & \cos(\theta+\alpha) \end{pmatrix} \cdot I_2 \cdot \begin{pmatrix} \cos\alpha & \sin\alpha \\ -\sin\alpha & \cos\alpha \end{pmatrix}$$
\end{block}

\end{frame}

%------------------------------------------------

\begin{frame}

\begin{block}{еҘҮејӮеҖјеҲҶи§Јзҡ„еә”з”ЁпјҡеӣҫеғҸеҺӢзј©}
еҲ©з”ЁSVDпјҢжҲ‘д»¬еҸҜд»ҘжҠҠдёҖдёӘз§©дёә$r$зҡ„$m\times n$зҹ©йҳө$M$еҶҷдҪң
$$M = \sigma_1\mathbf{u}_1\mathbf{v}_1^T + \cdots + \sigma_r\mathbf{u}_r\mathbf{v}_r^T.$$

иҖғиҷ‘$M$жҳҜдёҖе№…$m\times n$дёӘеғҸзҙ зҡ„еӣҫзүҮзҡ„жғ…еҶөгҖӮе°Ҫз®ЎеӣҫзүҮдёҖиҲ¬жҺҘиҝ‘ж»Ўз§©пјҢдҪҶ{\color{red}жңүж•Ҳз§©}еҫҲдҪҺгҖӮд№ҹе°ұжҳҜиҜҙпјҢеӯҳеңЁдёҖдёӘзӣёеҜ№дәҺ$r$еҫҲе°Ҹзҡ„$k$пјҢдҪҝеҫ—$\sigma_{k+1},\ldots,\sigma_{r}$йқһеёёе°ҸпјҢжҺҘиҝ‘дәҺ$0$пјҢеӣ жӯӨз”Ё
$$M_k := \sigma_1\mathbf{u}_1\mathbf{v}_1^T + \cdots + \sigma_k\mathbf{u}_k\mathbf{v}_k^T$$
еҸҜд»ҘеҫҲеҘҪең°иҝ‘дјј$M$пјҢеҗҢж—¶еҸӘйңҖиҰҒеӯҳеӮЁ$k(m+n+1)$дёӘж•°жҚ®пјҢиҖҢдёҚжҳҜеҺҹе§Ӣзҡ„$mn$дёӘж•°жҚ®гҖӮдәҺжҳҜжүҖйңҖеӯҳеӮЁзҡ„ж•°жҚ®еӨ§еӨ§еҮҸе°‘пјҢеӣҫеғҸеҫ—еҲ°дәҶеҺӢзј©гҖӮ
\end{block}

\end{frame}
