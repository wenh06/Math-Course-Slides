\section{基本定义}

%------------------------------------------------

\begin{frame}{Euclid 空间}

一维 $\longsquiggly$ 高维: ~~ 推广邻域、极限、连续、微分 (导数) 等概念

\pause
\vspace{1em}

\begin{itemize}[<+->]
\item 邻域: ~~ $O(x_0, \delta) = \{ x: ~ \lvert x - x_0 \rvert < \delta \}.$

\item (序列、函数) 极限 $\lim\limits_{n\to\infty} x_n = a$:
$$\forall \varepsilon > 0, ~ \text{存在正整数 $N$ 使得} ~ \forall n > N, ~ \text{有} ~ \lvert x_n - a \rvert < \varepsilon.$$

\item 函数 $f$ 在一点 $x_0$ 处连续:
$$\forall \varepsilon > 0, \exists \delta > 0, ~ \text{使得} ~ \forall \lvert x - x_0 \rvert < \delta, ~ \text{有} ~ \lvert f(x) - f(x_0) \rvert < \varepsilon.$$

\item 微分: $f(x_0 + h) - f(x_0) = L(x_0)(h) + o(\lvert h \rvert),$ 其中 $L(x_0)$ 是 $h$ 的线性函数.
\end{itemize}

\pause
\vspace{1em}

主要问题:

\begin{itemize}[<+->]
\item 推广 $\lvert \cdot \rvert$ 到高维
\item 推广线性函数到高维
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}{Euclid 空间}

\begin{Def}
$(\mathbb{R}, \langle \cdot, \cdot \rangle)$ 被称作 Euclid 空间 (内积空间), 其中
\begin{gather*}
\mathbb{R}^n = \left\{ (x_1, x_2, \dots, x_n) : ~ x_1, \dots, x_n \in \mathbb{R} \right\}, \\
\langle x, y \rangle = \sum\limits_{i=1}^n x_i y_i = x y^T.
\end{gather*}
\end{Def}

\pause

设 $A$ 为 $n$ 阶正定阵, 则
$$\langle x, y \rangle_A = xAy^T$$
也给出了内积空间的结构.

\pause
\vspace{0.5em}

一般地, $\mathbb{R}^n$ 上的内积指的是一个正定对称的双线性型
$$(x, y) : \mathbb{R}^n \times \mathbb{R}^n \longrightarrow \mathbb{R}$$
$(x, y) = xGy^T,$ $G$ 正定 (对称).

\end{frame}

%------------------------------------------------

\begin{frame}{Euclid 空间例子}

\begin{align*}
M_{n\times m}(\mathbb{R}) = \{ \text{所有} n\times m \text{实矩阵} \} & \rightarrow \mathbb{R}^{n\times m} \\
\begin{pmatrix}
    a_{11} & \cdots & a_{1m} \\
    \vdots & & \vdots \\
    a_{n1} & \cdots & a_{nm}
\end{pmatrix} & \mapsto (a_{11}, \cdots, a_{1m}, a_{21}, \cdots, a_{nm})
\end{align*}

\pause

那么 $\forall A, B \in M_{n\times m}(\mathbb{R}),$ 有
$$\langle A, B \rangle = \operatorname{tr} (AB^T).$$

\end{frame}

\section[赋范空间]{范数与赋范线性空间}

%------------------------------------------------

\begin{frame}{范数与赋范线性空间}

内积 $\longsquiggly$ 范数 \quad $\lVert x \rVert = \sqrt{\langle x, x \rangle}$

\pause

\begin{Def}
设 $V$ 为 $\mathbb{R}$ 上线性空间, $V$ 上的一个范数指的是
$$\lVert \cdot \rVert : ~ V \longrightarrow \mathbb{R}, ~~ \text{满足:}$$
\vspace{-2em}
\begin{enumerate}
    \item 正定性: $\lVert x \rVert \geqslant 0,$ 且 $\lVert x \rVert \geqslant 0$ 当且仅当 $x = 0;$
    \item 数乘相容性: $\forall \lambda \in \mathbb{R}, \lVert \lambda x \rVert = \lvert \lambda \rvert \lVert x \rVert;$
    \item 三角不等式: $\lVert x + y \rVert \leqslant \lVert x \rVert + \lVert y \rVert.$
\end{enumerate}
$(V, \lVert \cdot \rVert)$ 称为一个赋范线性空间.
\end{Def}

\pause

\begin{prop}
由内积诱导的映射 $x \mapsto \sqrt{\langle x, x \rangle} = \lVert x \rVert$ 是一个范数, 并且满足平行四边形等式
$$\lVert x + y \rVert^2 + \rVert x - y \rVert^2 = 2 ( \lVert x \rVert^2 + \lVert y \rVert^2 ).$$
\end{prop}

\end{frame}

%------------------------------------------------

\begin{frame}{范数与赋范线性空间}

\begin{block}{范数的例子}
$M_{n\times m}(\mathbb{R})$ 上由 $\langle A, B \rangle = \operatorname{tr} (AB^T)$ 对应的范数为
$$\lVert A \rVert_F = \sqrt{\operatorname{tr} (AA^T)} = \sqrt{\sum\limits_{i, j} \lvert a_{ij} \rvert^2}$$
称为矩阵的 Frobenius 范数.
\end{block}

\pause

矩阵范数还有很多, 例如算子范数
\vspace{-1em}
$$
A: ~ \mathbb{R}^{(m)} \rightarrow \mathbb{R}^{(n)}, ~~ x = \begin{pmatrix}
x_1 \\ \vdots \\ x_m
\end{pmatrix} \mapsto Ax,
$$
定义$\lVert A \rVert = \sup\limits_{x\neq 0} \dfrac{\lVert Ax \rVert}{\lVert x \rVert} = \sup\limits_{\lVert x \rVert = 1} \lVert Ax \rVert,$
\pause
或更一般的$\sup\limits_{x\neq 0} \dfrac{\lVert Ax \rVert_q}{\lVert x \rVert_p},$ 其中
$$\lVert x \rVert_p = \left( \lvert x_1 \rvert^p + \cdots + \lvert x_n \rvert^p \right)^{1/p} ~ \text{为 $p$ 范数}.$$

\pause

{\color{red}注意}, 不是所有范数都可以由某个内积所诱导,
\pause
例如 $p$ 范数, $p\neq 2.$

\end{frame}

%------------------------------------------------

\begin{frame}{范数的等价}

\begin{Def}
设 $\lVert \cdot \rVert_1, \lVert \cdot \rVert_2$ 是 $V$ 上的两个范数, 若存在正常数 $c_1, c_2$ 使得
$$c_1 \lVert x \rVert_1 \leqslant \lVert x \rVert_2 \leqslant c_2 \lVert x \rVert_1, ~ \forall x \in V,$$
则称 $\lVert \cdot \rVert_1$ 与 $\lVert \cdot \rVert_2$ 等价.
\end{Def}

\pause
\vspace{1em}

一个好消息是, $\mathbb{R}^n$ 上的所有范数都是等价的. 这将是在泛函分析课程中要证明的一个重要结论.

这个结论的好处在于, 在 $\mathbb{R}^n$ 中的 $x \to x_0,$ 可以用任意一个更加方便的范数来代替原来的范数.

\pause
\vspace{1em}

我们马上要用到两个范数等价的结论, 即向量的 $1$ 范数和 $2$ 范数等价, 证明也是非常简单的.

\fi

\end{frame}

\section[度量空间]{度量空间}

%------------------------------------------------

\begin{frame}{度量空间}

范数 $\longsquiggly$ 度量 \quad $d(x, y) = \lVert x - y \rVert$

\pause
\vspace{1em}

度量满足
\begin{enumerate}
\item 非负性与非退化性: $d(x, y) \geqslant 0;$ 且 $d(x ,y) = 0$ 当且仅当 $x = y;$
\item 对称性: $d(x, y) = d(y, x);$
\item 三角不等式: $d(x, y) \leqslant d(x, z) + d(y, z).$
\end{enumerate}

\pause
\vspace{1em}

值得注意的是, 度量的非负性 $d(x, y) \geqslant 0$ 可以由三角不等式推出 (取 $x = y$).

\pause
\vspace{1em}

不是所有度量都能由某个范数所诱导, 例如离散度量
$$
d(x, y) = \begin{cases}
1, & x \neq y, \\ 0, & x = y
\end{cases}
$$

\end{frame}

%------------------------------------------------

\begin{frame}{更多度量空间的例子}

闭区间 $[a, b] \subset \mathbb{R}$ 上连续函数空间 $C[a, b],$ 之前在学习函数项级数 (函数列) 一致收敛性时就已经接触过如下的度量:
$$d(f, g) = \sup\limits_{x \in [a, b]} \lvert f(x) - g(x) \rvert.$$
可以验证, 这是一个度量.

\pause
\vspace{1em}

$C[a, b]$ 上还可以赋予其他度量, 例如
$$d_p(f, g) = \left( \int_a^b \lvert f(x) - g(x) \rvert^p ~ \mathrm{d}x \right)^{1/p}.$$
由 Minkowski 不等式易知, 这是一个度量.
\pause
并且进一步还有 $d_{\infty}(f,g) = d(f,g).$

\end{frame}

\section{拓扑空间}

%------------------------------------------------

\begin{frame}{度量诱导的拓扑}

度量 $\longsquiggly$ 拓扑 \quad (定义开集、闭集)

\pause

\begin{Def}[开集]
集合 $E \subset \mathbb{R}^n$ 被称作开集, 若 $E$ 中所有的点都是{\color{red}内点}, 即 $\forall x \in E,$ {\color{red} 存在 $x$ 的邻域 $O(x, \delta), \delta > 0,$ 使得 $O(x, \delta) \subset E.$}
$$\text{邻域} ~~ O(x, \delta) = \{ y: ~ d(x, y) < \delta \}.$$
\end{Def}

\pause
\vspace{1em}

去心邻域
$$\mathring{O}(x, \delta) = \{ y: ~ 0 < d(x, y) < \delta \} = O(x, \delta) \setminus \{x\}.$$

\pause
\vspace{1em}

\begin{Def}[闭集]
集合 $E \subset \mathbb{R}^n$ 被称作闭集, 若它的补集 $\mathbb{R}^n \setminus E$ 是开集.
\end{Def}

\end{frame}

%------------------------------------------------

\begin{frame}{度量诱导的拓扑}

不是所有的拓扑都是由某个度量诱导的, 因为所有由度量诱导的拓扑都是 Hausdorff 的 (T2).

\pause
\vspace{1em}

T2: $\forall x \neq y,$ 存在 $x$ 的邻域 $U(x),$ $y$ 的邻域 $U(y),$ 使得 $U(x) \cap U(y) = \emptyset.$

\pause
\vspace{1em}

\begin{block}{所有由度量诱导的拓扑都是 Hausdorff 的证明}
任取 $x \neq y,$ 则 $d = d(x, y) > 0.$ 取 $x$ 的邻域 $U(x) = O(x, d/3),$ $y$ 的邻域 $U(y) = O(y, d/3).$ 假设 $U(x) \cap U(y) \neq \emptyset,$ 任取 $z \in U(x) \cap U(y),$ 那么有
$$d(x, z) < d/3, ~~ d(y, z) < d/3,$$
于是根据三角不等式有
$$d(x, y) \leqslant d(x, z) + d(y, z) < 2d/3 < d,$$
矛盾. 因此必须有$U(x) \cap U(y) = \emptyset.$
\end{block}

\end{frame}

%------------------------------------------------

\begin{frame}{其余部分类型的拓扑空间}

\begin{itemize}[<+->]
\item T1 (Fréchet):

$\forall x \neq y,$ 存在 $x$ 的邻域 $U(x)$ 使得 $y \not\in U(x),$ {\color{red}并且} 存在 $y$ 的邻域 $U(y)$ 使得 $x \not \in U(y).$

\vspace{1em}

\item T0 (Kolmogorov):

$\forall x \neq y,$ 存在 $x$ 的邻域 $U(x)$ 使得 $y \not\in U(x),$ {\color{red}或者} 存在 $y$ 的邻域 $U(y)$ 使得 $x \not\in U(y).$
\end{itemize}

\end{frame}

\begin{frame}{非 Hausdorff 拓扑空间的例子}

Zariski拓扑
\pause
$k^n = \operatorname{Spec} k[x_1, \dots, x_n],$ $k$为代数闭域. 更一般地, 考虑素谱
$$\operatorname{Spec} A = \{ A ~ \text{的所有素理想} ~ \mathfrak{p} \},$$
$A$ 为一个交换环. $\operatorname{Spec} A$ 的所有闭集为
$$V(\mathfrak{a}) = \{ \mathfrak{p} \in \operatorname{Spec} A:
\mathfrak{a} \subset \mathfrak{p} \}, ~~ \mathfrak{a} ~\text{为理想}.$$
\pause
若素理想 $\mathfrak{p}_1 \subsetneqq \mathfrak{p}_2,$ 那么任意包含点 $\mathfrak{p}_2$ 的开集必然也包含点 $\mathfrak{p}_1.$

\pause

这是因为, 设包含点 $\mathfrak{p}_2$ 的开集是某个 $V(\mathfrak{a})$ 的补集, 那么 $\mathfrak{a} \not\subset \mathfrak{p}_2,$ 从而必然有 $\mathfrak{a} \not\subset \mathfrak{p}_1,$ 于是该开集包含 $\mathfrak{p}_1.$

这种情况下, $\operatorname{Spec}A$ 是 T0 拓扑空间.

\pause
\vspace{0.5em}

若希望 $\operatorname{Spec}A$ 是 T1 拓扑空间, 则需要 $A$ 中所有可能的素理想链
$$\mathfrak{p}_0 \subsetneqq \mathfrak{p}_1 \subsetneqq \cdots \subsetneqq \mathfrak{p}_n$$
的最大下标 $n$ (所谓的 Krull 维数) 等于 $0,$ 也就是每个素理想都是极大理想.
\pause
或者考虑极大谱 $\operatorname{Max} A = \{ A ~ \text{的所有极大理想} \},$ 这是一个 T1 拓扑空间.

\fi

\end{frame}

\section[更多概念与性质]{Euclid 空间进一步的定义与性质}

%------------------------------------------------

\begin{frame}{收敛与极限}

\begin{Def}
设 $\{x_k\}_{k\in\mathbb{N}}$ 是 $\mathbb{R}^n$ 中的点列, 若存在点 $a \in \mathbb{R}^n,$ 使得 $\forall \varepsilon > 0,$ 存在正整数 $K,$ 使得 $\forall k > K,$ 有
$$d(x_k, a) = \lVert x_k - a \rVert < \varepsilon ~~ (\text{或}~ x_k \in O(a, \varepsilon)),$$
则称点列 $\{x_k\}_{k\in\mathbb{N}}$ 收敛于 $a,$ 记为 $\lim\limits_{k\to\infty} x_k = a,$ $a$ 称为 $\{x_k\}_{k\in\mathbb{N}}$ 的极限.
\end{Def}

\pause

\begin{thm}
$\lim\limits_{k\to\infty} x_k = a$ 当且仅当 $\forall 1 \leqslant i \leqslant n, \lim\limits_{k\to\infty} x_i^k = a_i,$ 其中 $x_k = (x_1^k, \dots, x_n^k).$
\end{thm}

\end{frame}

%------------------------------------------------

\begin{frame}{收敛与极限}

由不等式
$$d(x_k, a) = \lVert x_k - a \rVert = \sqrt{\sum\limits_{i=1}^n (x_i^k - a_i)^2} \begin{cases}
\geqslant ~~ \lvert x_i^k - a_i \rvert, ~~ \forall 1 \leqslant i \leqslant n \\
\leqslant ~~ \sum\limits_{i=1}^n \lvert x_i^k - a_i \rvert
\end{cases}$$
由夹逼准则即可知定理成立.

\pause
\vspace{1em}

以上再加上不等式 $\sum\limits_{i=1}^n \lvert x_i^k - a_i \rvert = \lVert x_k - a \rVert_1 \leqslant \sqrt{n} \lVert x_k - a \rVert$ 即可得 $1$ 范数与 $2$ 范数等价.
\pause
{\color{red}思考}: 如何得到与 $\infty$ 范数等价?

\end{frame}

%------------------------------------------------

\begin{frame}{Cauchy 收敛原理}

\begin{Def}[$\mathbb{R}^n$ 中的基本点列 (Cauchy 点列)]
设 $\{x_k\}_{k\in\mathbb{N}}$ 是 $\mathbb{R}^n$ 中的点列, 若对于任意 $\varepsilon > 0,$ 存在正整数 $K,$ 使得 $\forall k, \ell > K,$ 有
$$d(x_k, x_\ell) = \lVert x_k - x_\ell \rVert < \varepsilon,$$
则称点列 $\{x_k\}_{k\in\mathbb{N}}$ 是基本点列 (Cauchy 点列).
\end{Def}

\pause

\begin{thm}[Cauchy 收敛原理]
$\mathbb{R}^n$ 中的点列 $\{x_k\}_{k\in\mathbb{N}}$ 收敛的充要条件是 $\{x_k\}_{k\in\mathbb{N}}$ 是 $\mathbb{R}^n$ 中的 Cauchy 列.
\end{thm}

\pause

由不等式$
d(x_k, x_\ell) ~~ \begin{cases}
\geqslant ~~ \lvert x_i^k - x_i^\ell \rvert, ~~ \forall 1 \leqslant i \leqslant n \\
\leqslant ~~ \sum\limits_{i=1}^n \lvert x_i^k - x_i^\ell \rvert
\end{cases}
$
知 $\{x_k\}_{k\in\mathbb{N}}$ 是 $\mathbb{R}^n$ 中的 Cauchy 列的充要条件是 $\forall 1 \leqslant i \leqslant n, \{x_i^k\}_{k\in\mathbb{N}}$ 是 $\mathbb{R}$ 中的 Cauchy 列.

\end{frame}

%------------------------------------------------

\begin{frame}{Cauchy 收敛原理}

{\color{red}注意}, 对一般的度量空间, Cauchy 收敛原理不一定成立. Cauchy 收敛原理成立的度量空间被称作是完备的.

\vspace{1em}

例如在绝对值度量下, $\mathbb{R}$ 是完备的, 但 $\mathbb{Q}$ 不是.
\pause
正是由 $\mathbb{R}$ 的完备性, 我们有由划分对应的积分和给出的闭区间上有界函数 Riemann 可积的柯西收敛原理.

\pause
\vspace{1em}

又例如区间$[a, b]$ 上连续函数空间 $C[a, b],$ 在度量
$$d(f, g) = \sup\limits_{x \in [a, b]} \lvert f(x) - g(x) \rvert$$
下, 是一个完备的度量空间. 这正是一致收敛函数列极限函数连续性定理.

\end{frame}

%------------------------------------------------

\begin{frame}{Cauchy 收敛原理}

\begin{block}{思考}
\begin{itemize}[<+->]
\item $C[a, b]$ 在度量 $d_p(f, g) = \left( \int_a^b \lvert f(x) - g(x) \rvert^p ~ \mathrm{d}x \right)^{1/p}$ 下是否完备?
\item $[a, b]$ 上 Riemann 可积函数空间 $\mathfrak{R}[a, b]$ 在度量 $d, d_p$ 下是否完备?
\end{itemize}
\end{block}

\end{frame}

%------------------------------------------------

\begin{frame}{不完备的度量空间}

\begin{block}{结论1}
$C[a, b]$ 在度量 $d_p$ 下不完备.
\end{block}

\pause
\vspace{1em}

例如 $[a, b] = [-1, 1],$ 考虑函数列
$$f_k(x) = \begin{cases}
-1, & x \in [-1, -1/k), \\
kx, & x \in [-1/k, 1/k], \\
1, & x \in (1/k, 1].
\end{cases}$$
那么 $\lVert f_k - f_{k+s} \rVert_p \leqslant \left(2 \cdot \dfrac{1}{k}\right)^{1/p} \to 0, ~ (k \to \infty)$ 但极限函数 $f(x) = \operatorname{sgn}(x)$ 不在 $C[-1, 1]$ 中.

\end{frame}

%------------------------------------------------

\begin{frame}{不完备的度量空间}

\begin{block}{结论2}
$[a, b]$ 上 Riemann 可积函数空间 $\mathfrak{R}[a, b]$ 在度量 $d_p$ 下也不完备.
\end{block}

\pause
\vspace{1em}

例如 $[a, b] = [0, 1],$ 并将 $[0, 1]$ 上的所有有理数写为 $r_1, r_2, \dots,$ 考虑函数列
$$f_k(x) = \begin{cases}
1, & x \in \{r_1, r_2, \dots, r_k\}, \\
0, & \text{其他}.
\end{cases}$$
那么 $\lVert f_k - f_{k+s} \rVert_p \equiv 0,$ 但极限函数 Dirichlet 函数不在 $\mathfrak{R}[0, 1]$ 中.

\pause
\vspace{1em}

{\color{red}但是}, $\mathfrak{R}[a, b]$ 在度量 $d$ 下是完备的, 这正是一致收敛函数列极限函数可积性定理.

\end{frame}

%------------------------------------------------

\begin{frame}{更多点集拓扑相关的概念}

之前引入开集这一概念时, 我们介绍了内点的概念. 接下来, 我们介绍更多点集拓扑相关的概念.

\pause
\vspace{1em}

\begin{itemize}[<+->]
\item 集合 $E \subset \mathbb{R}^n$ 的内点 $x:$ ~~ $\exists O(x, \delta) \subset E.$ \par
集合 $E \subset \mathbb{R}^n$ 的内部 $E^0:$ ~~ $\{ x \in \mathbb{R}^n: ~ x ~ \text{为 $E$ 的内点} \}.$
\item 集合 $E \subset \mathbb{R}^n$ 的外点 $x:$ ~~ $\exists O(x, \delta) \subset \mathbb{R}^n \setminus E.$
\item 集合 $E \subset \mathbb{R}^n$ 的边界点 $x:$ ~~ $\forall O(x, \delta) \cap E \neq \emptyset,$ 但 $O(x, \delta) \not\subset E.$ \par
集合 $E \subset \mathbb{R}^n$ 的边界 $\partial E:$ ~~ $\{ x \in \mathbb{R}^n: ~ x ~ \text{为 $E$ 的边界点} \}.$
\item 集合 $E \subset \mathbb{R}^n$ 的孤立点 $x:$ ~~ $\exists O(x, \delta)$ 使得 $O(x, \delta) \cap E = \{x\}.$
\item 集合 $E \subset \mathbb{R}^n$ 的聚点 (极限点) $x:$ ~~ $\forall \mathring{O}(x, \delta), \mathring{O}(x, \delta) \cap E \neq \emptyset.$ \par
集合 $E \subset \mathbb{R}^n$ 的导集 $E':$ ~~ $\{ x \in \mathbb{R}^n: ~ x ~ \text{为 $E$ 的极限点} \}.$
\item 集合 $E \subset \mathbb{R}^n$ 的闭包 $\overline{E}:$ ~~ $E \cup E'.$
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}{聚点 (极限点)}

\begin{thm}
点 $x$ 是集合 $E$ 的聚点的充要条件是, 存在点列 $x_k \in E,$ $x_k \neq x,$ $k \in \mathbb{N}$ 使得 $\lim\limits_{k\to\infty} x_k = x.$
\end{thm}

\pause

$\Longrightarrow:$ 由定义, 取 $\delta_1 = 1,$ 则 $\mathring{O}(x, \delta_1) \cap E \neq \emptyset,$ 所以可以取到 $x_1 \in E, x_1 \neq x$ 使得 $x_1 \in \mathring{O}(x, \delta_1) \cap E.$

令 $\delta_2 = \dfrac{1}{2} d(x, x_1),$ 则 $\mathring{O}(x, \delta_2) \cap E \neq \emptyset,$ 所以可以取到 $x_2 \in E, x_2 \neq x$ 使得 $x_2 \in \mathring{O}(x, \delta_2) \cap E.$

于是, 可以归纳地构造出一个点列 $\{x_k\}_{k\in\mathbb{N}}$ 使得 $0 < d(x, x_k) < \dfrac{1}{2} d(x, x_{k-1}),$ 从而有 $\lim\limits_{k\to\infty} d(x_k, x) = 0,$ 即有 $\lim\limits_{k\to\infty} x_k = x.$

\pause
\vspace{0.5em}

$\Longleftarrow:$ 若 $\lim\limits_{k\to\infty} x_k = x,$ 且 $x_k \neq x,$ 则 $\forall \varepsilon > 0,$ 存在正整数 $K,$ 使得 $\forall k > K,$ 有 $0 < d(x_k, x) < \varepsilon,$ 即 $x_k \in \mathring{O}(x, \varepsilon) \cap E,$ 从而知 $\mathring{O}(x, \varepsilon) \cap E \neq \emptyset,$ 于是 $x$ 是 $E$ 的聚点.

\end{frame}

%------------------------------------------------

\begin{frame}{闭集}

\begin{prop}
集合 $E$ 的闭包 $\overline{E}$ 是闭集.
\end{prop}

\pause
\vspace{1em}

只要证明 $\mathbb{R}^n \setminus \overline{E}$ 是开集即可.
\pause
$\forall x \in \mathbb{R}^n \setminus \overline{E},$ 则 $x$ 不是 $E$ 的聚点, 即存在 $\mathring{O}(x, \delta) \cap E = \emptyset,$ 即 $\mathring{O}(x, \delta) \subset \mathbb{R}^n \setminus E,$ 于是
$$O(x, \delta) = \mathring{O}(x, \delta) \cup \{x\} \subset \mathbb{R}^n \setminus E.$$

\pause

考虑 $x$ 的邻域 $O(x, \delta / 2) \subset O(x, \delta) \subset \mathbb{R}^n \setminus E,$
假设 $O(x, \delta / 2)$ 含有 $E$ 的某个聚点 $z,$ 则 $\mathring{O}(z, \delta / 4) \cap E \neq \emptyset.$
但另一方面显然又有 $\mathring{O}(z, \delta / 4) \subset O(x, \delta),$ 矛盾. 所以 $O(x, \delta / 2) \cap \overline{E} = \emptyset,$
即 $O(x, \delta / 2) \subset \mathbb{R}^n \setminus \overline{E}.$ 因此 $\mathbb{R}^n \setminus \overline{E}$ 是开集.

\end{frame}

%------------------------------------------------

\begin{frame}{闭集}

\begin{prop}
$E$ 为闭集的充要条件是 $E = \overline{E}.$
\end{prop}

\pause
\vspace{1em}

$\Longleftarrow:$ 由上一命题可直接得到.

\pause
\vspace{0.5em}

$\Longrightarrow:$ 只要证明 $E$ 的任意聚点 $x$ 都在 $E$ 中即可.

\pause

假设聚点 $x \not\in E,$ 则 $x \in \mathbb{R}^n \setminus E,$ 由闭集定义知 $\mathbb{R}^n \setminus E$ 为开集, 即存在 $O(x, \delta) \subset \mathbb{R}^n \setminus E,$ 并进一步有 $\mathring{O}(x, \delta) \subset \mathbb{R}^n \setminus E,$ 从而 $\mathring{O}(x, \delta) \cap E = \emptyset,$ 即 $x$ 不是 $E$ 的聚点, 矛盾.

\end{frame}

%------------------------------------------------

\begin{frame}{开集}

\begin{prop}
我们定义的点 $x$ 的邻域 $O(x, \delta)$ 是开集.
\end{prop}

\pause

\begin{proof}
$\forall y \in O(x, \delta),$ 则 $d(x, y) < \delta,$ 取 $\varepsilon = (\delta - d(x, y)) / 2 > 0.$ 任取 $z \in O(y, \varepsilon),$ 则 $d(y, z) < \varepsilon,$ 由三角不等式知
$$d(x, z) \leqslant d(x, y) + d(y, z) < d(x, y) + \varepsilon = (\delta + d(x, y) / 2 < \delta,$$
即 $z \in O(x, \delta),$ 从而 $O(y, \varepsilon) \subset O(x, \delta),$ 即 $O(x, \delta)$ 是开集.
\end{proof}

% \pause

% 对于 $\mathbb{R}^1,$ 开集的结构比较简单, 任意开集都可以表示为之多可数个开区间的并集. 这个结论我们学习后续课程的时候会证明.

\pause

\begin{block}{一些重要开集的例子}
\begin{itemize}[<+->]
\item $n$ 维开球 $B(a, r) = \{ x: ~ d(x, a) < r \} = \{ a: ~ \lVert x - a \rVert < r \}.$
\item $n$ 维开矩体 $Q = \{ x: ~ a_i < x_i < b_i, ~ 1 \leqslant i \leqslant n \}.$
\end{itemize}
\pause
在 $n = 1$ 的情况下, 它们都是开区间.
\end{block}

\end{frame}

\begin{frame}{一般拓扑空间中的开集与闭集}

对于一个一般空间 $V,$ 它的开集指的是满足以下条件的子集族 $\mathscr{T}:$
\begin{enumerate}
\item $V$ 本身和空集 $\emptyset$ 都属于 $\mathscr{T}$ (都是开集);
\item 任意多个开集的并集仍然属于 $\mathscr{T}$ (是开集);
\item 有限多个开集的交集仍然属于 $\mathscr{T}$ (是开集).
\end{enumerate}
给定了这样一个子集族 $\mathscr{T},$ 就定义了 $V$ 上的一个拓扑空间的结构.
\pause
我们下面验证之前定义的开集满足这些条件.

\vspace{1em}

\begin{thm}
(1). 设 $\{ E_{\alpha} \}_{\alpha \in I}$ 是 $\mathbb{R}^n$ 上的一族开集, 则 $\bigcup\limits_{\alpha \in I} E_{\alpha}$ 仍然是开集. \par
(2). 若 $E_1, E_2, \dots, E_n$ 是 $\mathbb{R}^n$ 上的有限多个开集, 则 $\bigcap\limits_{i=1}^n E_i$ 仍然是开集.
\end{thm}

\end{frame}

%------------------------------------------------

\begin{frame}{定理的证明}

(1). $\forall x \in \bigcup\limits_{\alpha \in I} E_{\alpha},$ 则存在 $\alpha_0 \in I$ 使得 $x \in E_{\alpha_0},$ 由 $E_{\alpha_0}$ 是开集知存在 $O(x, \delta) \subset E_{\alpha_0} \subset \bigcup\limits_{\alpha \in I} E_{\alpha},$ 即 $\bigcup\limits_{\alpha \in I} E_{\alpha}$ 是开集.

\pause

(2). $\forall x \in \bigcap\limits_{i=1}^n E_i,$ 则 $\forall 1 \leqslant i \leqslant n, x \in E_i,$ 由 $E_i$ 是开集知存在 $O(x, \delta_i) \subset E_i,$ 取 $\delta = \min\limits_{1 \leqslant i \leqslant n} \delta_i,$ 则 $\forall 1 \leqslant i \leqslant n, O(x, \delta) \subset E_i,$ 即 $\bigcap\limits_{i=1}^n E_i$ 是开集.

\pause
\vspace{1em}

关于闭集, 我们对应地有
\begin{enumerate}
\item $V$ 本身和空集 $\emptyset$ 都是闭集;
\item 任意多个闭集的交集仍然是闭集;
\item 有限多个闭集的并集仍然是闭集.
\end{enumerate}

\pause
\vspace{1em}

一般来说, $\mathbb{R}^n$ 中任意多开集的交不一定仍是开集,
\pause
例如 $(1 - 1 / k, 3 + 1 / k).$

\pause

$\mathbb{R}^n$ 中任意多闭集的并不一定仍是闭集,
\pause
例如 $[1 + 1 / k, 3 - 1 / k].$

\end{frame}

\section[基本定理]{Euclid 空间上的基本定理}

\begin{frame}{闭矩体套定理}

\begin{thm}[闭矩体套定理]
设 $\{ \Delta_k \}$ 是 $\mathbb{R}^n$ 中的一列闭矩体, $\Delta_k = \{ x \in \mathbb{R}^n: ~ a_i^k \leqslant x_i \leqslant b_i^k, ~ 1 \leqslant i \leqslant n \},$ 且满足
$$\Delta_1 \supset \Delta_2 \supset \cdots \supset \Delta_k \supset \cdots,$$
或者写为
$$a_i^1 \leqslant a_i^2 \leqslant \cdots \leqslant a_i^k \leqslant \cdots \leqslant b_i^k \leqslant \cdots \leqslant b_i^2 \leqslant b_i^1, ~~ \forall 1 \leqslant i \leqslant n,$$
并且 $\lim\limits_{k\to\infty} (b_i^k - a_i^k) = 0, \forall 1 \leqslant i \leqslant n,$ 则存在唯一的点 $x \in \mathbb{R}^n,$ 使得 $x \in \bigcap\limits_{k=1}^{\infty} \Delta_k,$ 并且满足
$$\lim\limits_{k\to\infty} a_i^k = \lim\limits_{k\to\infty} b_i^k = x_i, ~~ \forall 1 \leqslant i \leqslant n.$$
\end{thm}

\end{frame}

%------------------------------------------------

\begin{frame}{Cantor 闭区域套定理}

\ifHandout
% do nothing
\else
\markEndOfClass{35}
\fi

闭矩体套定理的证明只要对每一维度 (坐标) 应用闭区域套定理即可.

\pause
\vspace{1em}

闭矩体套定理的关键在于“闭集”和“序列包含关系”以及“直径趋于零”. 我们可以将这个定理推广到更一般的情况.

\pause
\vspace{1em}

\begin{thm}[Cantor 闭区域套定理]
设 $\{ E_k \}$ 是 $\mathbb{R}^n$ 中的一列非空闭集, 且满足
$$E_1 \supset E_2 \supset \cdots \supset E_k \supset \cdots,$$
以及 $\lim\limits_{k\to\infty} \operatorname{diam} E_k = 0,$ 其中 $\operatorname{diam} E_k = \sup\limits_{x, y \in E_k} \lVert x - y \rVert,$ 则存在唯一的点 $x \in \mathbb{R}^n,$ 使得 $x \in \bigcap\limits_{k=1}^{\infty} E_k.$
\end{thm}

\end{frame}

%------------------------------------------------

\begin{frame}{Cantor 闭区域套定理证明}

首先证明 $x$ 的存在性, 即 $\bigcap\limits_{k=1}^{\infty} E_k \neq \emptyset.$

\pause

由于每个 $E_k$ 都是非空闭集, 于是可以取到点列 $x_k \in E_k.$ 由于对任意 $p \geqslant 1,$ 有 $E_{k+p} \subset E_k,$
于是 $x_{k+p} \in E_{k+p} \subset E_k,$ 从而有 $d(x_k, x_{k+p}) \leqslant \operatorname{diam} E_k,$
这表明 $\{ x_k \}_{k\in\mathbb{N}}$ 是 $\mathbb{R}^n$ 中的一个 Cauchy 列, 从而存在 $x \in \mathbb{R}^n,$ 使得 $\lim\limits_{k\to\infty} x_k = x.$

\pause

由于对任意 $p \geqslant 1,$ 有 $x_{k+p} \in E_k,$ 那么 $x$ 是每个闭集 $E_k$ 的聚点,
从而 $x \in \overline{E}_k = E_k,$ 即 $x \in \bigcap\limits_{k=1}^{\infty} E_k.$

\pause
\vspace{1em}

接下来证明 $x$ 的唯一性.

\pause

假设存在另一点 $y \in \bigcap\limits_{k=1}^{\infty} E_k,$ 则 $y \in E_k,$
从而 $d(x, y) \leqslant \operatorname{diam} E_k$ 对所有 $k$ 成立, 从而 $d(x, y) = 0,$ 即 $x = y.$

\end{frame}

\begin{frame}{Bolzano-Weierstra\ss 定理}

\begin{thm}[Bolzano-Weierstra\ss 定理]
设 $\{ x_k \}_{k\in\mathbb{N}}$ 是 $\mathbb{R}^n$ 中的有界点列, 则 $\{ x_k \}_{k\in\mathbb{N}}$ 必有收敛子列.
\end{thm}

\pause
\vspace{1em}

\begin{proof}
由于 $\{ x_k \}_{k\in\mathbb{N}}$ 是 $\mathbb{R}^n$ 中的有界点列, 于是对每一个坐标 $x_i$ 的数列 $\{ x_i^k \}$ 都是有界的.

\pause

对数列 $\{ x_1^k\}$ 应用一维的 Bolzano-Weierstra\ss 定理, 可以找到一个收敛子列 $\{ x_1^{k_j} \}$ 收敛到 $a_1.$

考虑数列 $\{ x_2^k\}$ 的相应子列 $\{ x_2^{k_j} \},$ 它也是有界的, 于是可以找到一个收敛子列 $\{ x_2^{k_{j_\ell}} \}$ 收敛到 $a_2.$
$\{ x_1^k\}$ 相应的子列 $\{ x_1^{k_{j_\ell}} \}$ 是收敛数列 $\{ x_1^{k_j} \}$ 的子列, 于是 $\{ x_1^{k_{j_\ell}} \}$ 也收敛到 $a_1.$

\pause

于是可以归纳地找到  $\{ x_k \}_{k\in\mathbb{N}}$ 的一个子列, 收敛到 $a = (a_1, \dots, a_n).$
\end{proof}

\end{frame}

\section{紧集}

%------------------------------------------------

\begin{frame}{紧集}

我们要引入一个重要的概念, 紧集. 紧性是拓扑空间一个重要性质.

\pause
\vspace{1em}

\begin{Def}[开覆盖]
设 $E$ 是拓扑空间 $V$ 中的一个子集, 若存在一族开集 $\{ G_{\alpha} \}_{\alpha \in I}$ 使得 $E \subset \bigcup\limits_{\alpha \in I} G_{\alpha},$ 则称 $\{ G_{\alpha} \}_{\alpha \in I}$ 是 $E$ 的一个开覆盖.
\end{Def}

\pause

\begin{Def}[紧集]
设 $E$ 是拓扑空间 $V$ 中的一个子集, 若 $E$ 的任意开覆盖 $\{ G_{\alpha} \}_{\alpha \in I}$ 都有有限子覆盖, 即存在有限个 $G_{\alpha_1}, \dots, G_{\alpha_p}$ 使得 $E \subset \bigcup\limits_{i=1}^p G_{\alpha_i},$ 则称 $E$ 是紧集.
\end{Def}

\pause

\begin{Def}[局部紧空间]
拓扑空间 $V$ 称为局部紧空间, 若 $V$ 中的每一点都有一个紧邻域, 即对每个 $x \in V,$ 存在 $x$ 的邻域 $U(x)$ 以及紧集 $K$ 使得 $x \in U(x) \subset K.$
\end{Def}

\end{frame}

%------------------------------------------------

\begin{frame}{单点紧化}

带有通常拓扑的 $\mathbb{R}^n$ 不是紧空间, 但是是一个局部紧空间. 对于一个一般的非紧的局部紧空间拓扑空间, 我们可以通过添加一个点来得到一个紧空间.

\pause
\vspace{1em}

设 $V$ 是一个局部紧空间, 我们可以定义 $V$ 的单点紧化 $V^* = V \cup \{ \infty \},$ 其中 $\infty$ 是一个新的点.
$V^*$ 中的开集由以下两类集合构成:

\begin{itemize}
\item $V$ 中的开集;
\item $V^* \setminus CK = (V \setminus CK) \cup \{ \infty \},$ 其中 $CK$ 是 $V$ 中的 (闭的) 紧集.
\end{itemize}

\pause
\vspace{1em}

很容易验证, 以上定义的开集满足拓扑空间中开集需要满足的三个条件, 也就是说, 以上给出了 $V^*$ 的一个拓扑空间结构.

并且可以验证, $V^*$ 是一个紧的 Hausdorff 空间.

\pause
\vspace{1em}

紧集有一系列良好的性质, 接下来我们来讨论有关紧集的一些基本的结论.

\ifHandout
\addTextWatermark[2.5]{本页内容不作为考试要求}
\fi

\end{frame}

%------------------------------------------------

\begin{frame}{Heine-Borel 定理}

\begin{thm}[Heine-Borel 定理]
$\mathbb{R}^n$ 中的点集 $E$ 是紧集的充要条件是 $E$ 是有界闭集.
\end{thm}

\pause

$\Longrightarrow:$ 若 $E$ 是紧集. 首先, 证明它是有界的.

\pause

我们取一个特殊的开覆盖, 即 $\{ O(x, 1) \}_{x \in E}.$ 由于 $E$ 是紧集, 于是存在有限个点 $x_1, \dots, x_p \in E,$ 使得 $E \subset \bigcup\limits_{i=1}^p O(x_i, 1),$ 于是 $E$ 是有界的.

\pause
\vspace{0.5em}

接下来, 证明 $E$ 是闭集.

\pause

用反证法. 假设 $E$ 不是闭集, 则存在 $E$ 的聚点 $x,$ 且 $x \not\in E.$ 于是
$$G_m = \mathbb{R}^n \setminus \overline{O(x, 1/m)} = \left\{ y \in \mathbb{R}^n: ~ d(x, y) > \dfrac{1}{m} \right\}, ~~ m \in \mathbb{N}$$
构成了 $E$ 的一个开覆盖. 下面证明 $\{ G_m \}$ 没有 $E$ 的有限子覆盖.

\pause

由于 $x$ 是 $E$ 的聚点, 即存在 $x_k \in E, x_k \neq x,$ 且 $\lim\limits_{k\to\infty} x_k = x,$ 于是对任意 $m,$ 存在 $K_m$ 使得 $d(x, x_k) < 1/m, \forall k > K_m,$ 从而 $x_k \not\in G_m, \forall k > K_m.$ 这表明每个 $G_m$ 都只含有有限个 $x_k,$ 从而 $\{ G_m \}$ 的任意有限子覆盖都只含有有限个 $x_k,$ 从而不能覆盖 $E,$ 矛盾.

\end{frame}

%------------------------------------------------

\begin{frame}{Heine-Borel 定理}

$\Longleftarrow:$ 若 $E$ 是有界闭集. 我们用反证法证明 $E$ 是紧集.

\pause
\vspace{0.5em}

假设 $E$ 不是紧集, 则存在 $E$ 的一个开覆盖 $\{ G_{\alpha} \}_{\alpha \in I},$ 使得任意有限个 $G_{\alpha_i}$ 不能覆盖 $E.$
由于 $E$ 是有界的, 于是存在一个有界闭矩体 $\Delta_{(0)} = [-M, M]^n,$ 使得 $E \subset \Delta_{(0)}.$
$\Delta_{(0)}$ 可以等分为 $2^n$ 个有界闭矩体 $\Delta_1, \dots, \Delta_{2^n}.$
这 $2^n$ 个有界闭矩体中至少有一个, 记为 $\Delta_{(1)},$ 使得 $\Delta_{(1)} \cap E$ 不能被 $\{ G_{\alpha} \}_{\alpha \in I}$ 的有限个元素覆盖.
于是我们取 $\Delta_{(1)}$ 作为新的有界闭矩体, 重复上述过程, 于是我们可以得到非空闭区域套
$$E = E \cap \Delta_{(0)} \supset E \cap \Delta_{(1)} \supset \cdots \supset E \cap \Delta_{(k)} \supset \cdots,$$
使得 $\Delta_{(k)}\cap E$ 不能被 $\{ G_{\alpha} \}_{\alpha \in I}$ 的有限个元素覆盖, 并且 $\lim\limits_{k\to\infty} \operatorname{diam} E \cap \Delta_{(k)} = 0.$

\pause

由 Cantor 闭区域套定理, 存在唯一的点 $x \in \bigcap\limits_{k=1}^{\infty} (E \cap \Delta_{(k)}).$ 任取 $G_{\alpha}$ 使得 $x \in G_{\alpha},$ 由于 $G_{\alpha}$ 是开集, 于是存在 $O(x, \delta) \subset G_{\alpha}.$ 由于 $\lim\limits_{k\to\infty} \operatorname{diam} E \cap \Delta_{(k)} = 0,$ 故 $\exists K,$ 使得 $E \cap \Delta_{(K)} \subset O(x, \delta) \subset G_{\alpha},$ 矛盾.

\end{frame}

%------------------------------------------------

\begin{frame}{$\mathbb{R}^n$ 中紧集的等价刻画}

我们再加上集合 $E$ 是紧集的一个 (等价) 条件:
\begin{center}
$E$ 的任意无限子集在 $E$ 中必有聚点.
\end{center}

\pause
\vspace{1em}

我们证明它与 $E$ 是有界闭集的等价性. 设 $S \subset E$ 是 $E$ 的无限子集.

\pause
\vspace{1em}

假设 $E$ 是有界闭集, 那么从 $S$ 中任选一个可列子集 $\{ x_k \},$ 那么它也构成了一个没有重复元素的点列,
由 Bolzano-Weierstra\ss 定理知 $\{ x_k \}_{k\in\mathbb{N}}$ 必有收敛子列. 可以不妨设 $\{ x_k \}_{k\in\mathbb{N}}$ 本身就是收敛的, 假设收敛到 $x,$ 那么任取 $\varepsilon > 0,$ 存在正整数 $K,$ 使得 $d(x_k, x) < \varepsilon, \forall k > K,$ 即 $x_k \in O(x, \varepsilon)$ 对所有 $k > K$ 成立, 从而 $\mathring{O}(x, \varepsilon) \cap S \neq \emptyset,$ 即 $x$ 是 $S$ 的聚点. 由于 $E$ 是闭集, 从而 $x \in E.$

\pause
\vspace{1em}

另一个方向, 只要证明 $E$ 有界集即可. 用反证法, 假设 $E$ 无界, 那么 $\forall k \in \mathbb{N},$ 存在 $x_k \in E,$ 使得 $\lVert x_k \rVert > \max\{k, \lVert x_{k-1} \rVert + 1\}$ (约定 $x_0$ 为 $E$ 中任意一点), 从而 $\{ x_k \}_{k\in\mathbb{N}}$ 是 $E$ 的无限子集, 但是它没有聚点, 矛盾.

\end{frame}

%------------------------------------------------

\begin{frame}{$\mathbb{R}^n$ 中紧集的等价刻画}

总结起来, 就是有如下的定理

\begin{thm}
设 $E \subset \mathbb{R}^n,$ 那么以下三个命题等价
\begin{enumerate}
\item $E$ 是有界闭集;
\item $E$ 是紧集;
\item $E$ 的任意无限子集在 $E$ 中必有聚点.
\end{enumerate}
\end{thm}

\end{frame}
