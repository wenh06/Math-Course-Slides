\section[定义]{上、下极限的定义}

%------------------------------------------------

\begin{frame}{数列的极限点}

\begin{Def}
在有界数列 $\{x_n\}$ 中, 若存在子列 $\{x_{n_k}\},$ 使得 $\lim\limits_{k \to \infty} x_{n_k} = \xi,$ 则称 $\xi$ 为数列 $\{x_n\}$ 的一个\alert{极限点}.
\end{Def}

\pause

极限点的概念可以等价地定义为: $\xi$ 的任意邻域中都包含 $\{x_n\}$ 中的无穷多项.记
\[E = \{ \xi\in\mathbb{R} ~:~ \xi \text{ 是 $\{x_n\}$ 的极限点} \},\]
对于有界数列, 由于它总有收敛子列, 因此 $E$ 非空.

\begin{attnblock}
数列极限点的概念要与点集拓扑中极限点的概念相区别. 点集拓扑中, 一个点 $\xi$ 是集合 $S$ 的极限点, 若 $\xi$ 的任意去心邻域与 $S$ 的交非空.

\pause

二者的唯一区别在于, 当 $\xi$ 本身在 $\{x_n\}$ 中出现无穷多次时, $\xi$ 是 $\{x_n\}$ 的极限点, 但可能只是相应点集的孤立点.

\pause

数列极限点与点集拓扑中附着点 (adhere point) 的概念更接近.
\end{attnblock}

\end{frame}

%------------------------------------------------

\begin{frame}{极限点集 $E$ 的性质}

% $\{x_n\}$ 极限点集 $E = \{ \xi\in\mathbb{R} ~:~ \xi \text{ 是 $\{x_n\}$ 的极限点} \}$ 有如下性质

\begin{thm}
\begin{center}
$E$ 的上确界 $H$ 与下确界 $h$ 均属于 $E.$
\end{center}
\end{thm}

\pause
\vspace{1em}

证明: 由于 $H = \sup E,$ 那么 $\forall~\varepsilon > 0,$ 存在 $\xi \in E$ 使得 $0 \leqslant H - \xi < \varepsilon.$ 又由于 $\xi \in E$ 为 $\{x_n\}$ 的极限点, 因此 $\xi$ 的邻域 $(\xi - \delta, \xi + \delta)$ 中包含 $\{x_n\}$ 的无穷多项, 这里 $\delta = \frac{\xi - (H - \varepsilon)}{2}.$

\begin{center}

\node[below, yshift=-4] at (he) {\smaller $H-\varepsilon$};

\node[below, yshift=-4] at (h) {\smaller $H$};

\draw (he) arc (180:140:0.2);
\draw (he) arc (180:220:0.2);

\end{tikzpicture}
\end{center}

由于 $O(\xi, \delta) \subset O(H, \varepsilon),$ 因此 $H$ 的 $\varepsilon$ 邻域也包含 $\{x_n\}$ 的无穷多项. 由于 $\varepsilon$ 是任取的, 因此 $H$ 也是 $\{x_n\}$ 的极限点, 即 $H \in E.$

\pause

\begin{cor}
$E$ 存在最大值与最小值, 分别为上确界 $H$ 与下确界 $h.$
\end{cor}

\end{frame}

%------------------------------------------------

\begin{frame}{上、下极限的定义}

\begin{Def}
$E$ 的最大值 $H = \max E$ 称为数列 $\{x_n\}$ 的上极限, 记为 $\varlimsup\limits_{n\to\infty} x_n;$

$E$ 的最小值 $h = \min E$ 称为数列 $\{x_n\}$ 的下极限, 记为 $\varliminf\limits_{n\to\infty} x_n.$
\end{Def}

\pause
\vspace*{1em}

上、下极限更普遍的定义是
\begin{equation*}
\varlimsup_{n\to\infty} x_n = \newinf_{n \geqslant 1} \sup_{k \geqslant n} x_k; \quad \varliminf\limits_{n\to\infty} x_n = \sup_{n \geqslant 1} \newinf_{k \geqslant n} x_k.
\end{equation*}

\pause

接下来我们证明这两个定义等价. 我们证明 $\displaystyle H = \newinf_{n \geqslant 1} \sup_{k \geqslant n} x_k,$ 另一个的证明是类似的. 记 $a_n = \sup\limits_{k \geqslant n} x_k,$ 那么 $\{ a_n \}$ 是一个不增的数列.

\end{frame}

%------------------------------------------------

\begin{frame}{上、下极限等价定义}

{
\smaller[1]

$1^\circ$ $H \leqslant \newinf_{n \geqslant 1} \sup_{k \geqslant n} x_k:$

\pause

任取 $\xi \in E,$ 那么存在 $\{x_n\}$ 的子列 $\{x_{n_k}\},$ 使得 $\lim\limits_{k\to\infty} x_{n_k} = \xi.$ $\forall ~ n \in \mathbb{N},$ 存在正整数 $K,$ 使得任意 $k \geqslant K,$ $n_k \geqslant n,$ 于是 $x_{n_k} \leqslant a_n = \sup\limits_{k \geqslant n} x_k$ 对所有 $k \geqslant K$ 成立. 对 $k$ 取极限即有 $\xi \leqslant a_n.$
\pause
由于这是对所有 $n$ 都成立的, 于是有 $\xi \leqslant \inf\limits_{n \geqslant 1}a_n.$ 由于 $\xi$ 是任取的, 所以 $\displaystyle H = \max E \leqslant \inf_{n \geqslant 1}a_n = \newinf_{n \geqslant 1} \sup_{k \geqslant n} x_k.$

\pause
\vspace*{0.5em}

$2^\circ$ $H \geqslant \newinf_{n \geqslant 1} \sup_{k \geqslant n} x_k:$

\pause

记 $\displaystyle I = \newinf_{n \geqslant 1} \sup_{k \geqslant n} x_k,$ 那么由下确界的性质, $\forall~\varepsilon > 0,$ 存在正整数 $N(\varepsilon),$ 使得 $\displaystyle I \leqslant \sup_{k \geqslant N(\varepsilon)} x_k = a_{N(\varepsilon)} < I + \varepsilon.$ 由上确界的性质, 存在另一个正整数 $n(\varepsilon) \geqslant N(\varepsilon),$ 使得 $\displaystyle a_{N(\varepsilon)} \geqslant x_{n(\varepsilon)} > a_{N(\varepsilon)} - \varepsilon.$

\begin{center}
\begin{tikzpicture}[scale=1.3]

\draw[-Stealth] (-1,0) -- (7,0);
\draw[thick] (0, 0) node[below] {\smaller $I - \varepsilon$} -- (0, 0.1);
\draw[thick] (1, 0) node[below] {\smaller $a_{N(\varepsilon)} - \varepsilon$} -- (1, 0.1);
\draw[thick] (3, 0) node[below] {\smaller $I$} -- (3, 0.1);
\draw[thick] (4, 0) node[below] {\smaller $a_{N(\varepsilon)}$} -- (4, 0.1);
\draw[thick] (6, 0) node[below] {\smaller $I + \varepsilon$} -- (6, 0.1);
\node at (2.5, 0) {{\color{red} \XSolid}};
\node[above] at (2.5, 0.1) {{\color{red} $x_{n(\varepsilon)}$}};
\end{tikzpicture}
\end{center}

有 $I - \varepsilon < x_{n(\varepsilon)} < I + \varepsilon.$ 分别取 $\varepsilon = 1, \frac{1}{2}, \frac{1}{3}, \dots, \frac{1}{t}, \dots,$ 可得子列 $\{x_{n_t}\}$ 收敛到 $I.$ 这里 $n_t = n(\frac{1}{t}),$ (注意, 可以做到下标 $n_t$ 不重复) 于是 $E \ni I \leqslant \max E.$

}

\end{frame}

%------------------------------------------------

\begin{frame}{由上、下极限给出的有界数列收敛充要条件}

\begin{thm}
设 $\{x_n\}$ 是有界数列, 则其收敛的充要条件为 $\displaystyle \varlimsup_{n\to\infty} x_n = \varliminf_{n\to\infty} x_n.$
\end{thm}

\pause
\vspace*{-1em}

\begin{align*}
\text{$\{x_n\}$ 收敛} ~~ & \implies ~~ \text{所有子列都收敛到同一个值} \\
& \implies ~~ \text{$E$ 只含有一个元素} \\
& \implies ~~ \varlimsup_{n\to\infty} x_n = \max E = \min E= \varliminf_{n\to\infty} x_n
\end{align*}

\pause
\vspace*{-0.5em}

\begin{align*}
\text{$\{x_n\}$ 不收敛} ~~ & {\color{red} \implies} ~~ \text{存在至少两个子列收敛到不同的值} \\
& \implies ~~ \text{$E$ 只至少含有两个个元素} \\
& \implies ~~ \varlimsup_{n\to\infty} x_n = \max E > \min E= \varliminf_{n\to\infty} x_n
\end{align*}

\pause

\alert{注意:} 红色的箭头 ${\color{red} \implies}$ 用到了如下的结论: 若\alert{有界}数列 $\{x_n\}$ 所有收敛子列都收敛到同一个值, 那么 $\{x_n\}$ 本身是收敛的.

\end{frame}

%------------------------------------------------

\begin{frame}{无界数列的上、下极限}

可以将极限点的概念 (借由``无穷的邻域'') 扩充至 $\pm \infty$ 的情况.

\begin{Def}
在数列 $\{x_n\}$ 中, 若存在子列 $\{x_{n_k}\},$ 使得 $\lim\limits_{k\to\infty} x_{n_k} = \xi,$ \alert{$-\infty \leqslant \xi \leqslant +\infty$}, 则称 $\xi$ 为数列 $\{x_n\}$ 的一个\alert{极限点}.
\end{Def}

\pause

\begin{remark}
用``无穷的邻域'', 可以将上述定义写为:
\begin{center}
$\xi$ 的任意邻域中都包含 $\{x_n\}$ 中的无穷多项.
\end{center}
\end{remark}

\pause

\begin{thm}
$\lim\limits_{n\to\infty} x_n$ 存在 (有限或 $+\infty$ 或 $-\infty$) 的充要条件为 $\varlimsup\limits_{n\to\infty} x_n = \varliminf\limits_{n\to\infty} x_n.$
\end{thm}
\end{frame}

%------------------------------------------------

\begin{frame}{无界数列的上、下极限}

\begin{thm}
$\lim\limits_{n\to\infty} x_n$ 存在 (有限或 $+\infty$ 或 $-\infty$) 的充要条件为 $\varlimsup\limits_{n\to\infty} x_n = \varliminf\limits_{n\to\infty} x_n.$
\end{thm}

\pause

对 $+\infty$ 的情况进行证明.

$\implies:$ 由于 $\lim\limits_{n\to\infty} x_n = +\infty,$ 那么 $\forall~M > 0,$ 存在正整数 $N,$ 当 $n > N$ 时, 有 $x_n > M.$ 于是 $\inf_{k \geqslant n} x_k > M$ 对所有 $n > N$ 成立, 因此 $\varliminf\limits_{n\to\infty} x_n = \sup\limits_{n \geqslant 1} \inf\limits_{k \geqslant n} x_k \geqslant M.$ 由于 $M$ 是任取的, 因此 $\varliminf\limits_{n\to\infty} x_n = +\infty.$

\pause
\vspace{1em}

$\Longleftarrow:$ 若 $\varliminf\limits_{n\to\infty} x_n = \sup_{n\geqslant 1}\inf_{k \geqslant n} x_k = +\infty,$ 用反证法证明, 对于任意 $M > 0,$ 存在正整数 $N,$ 当 $n > N$ 时, 有 $x_n > M:$ 假设对某个 $M_0 > 0,$ 存在无穷多个 $x_n \leqslant M_0,$ 记这些下标为 $\{n_k\},$ 则 $\inf\limits_{k \geqslant n_k} x_k \leqslant x_{n_k} \leqslant M_0,$ 对所有 $k$ 成立, 因此 $\varliminf\limits_{n\to\infty} x_n = \sup\limits_{n \geqslant 1} \inf\limits_{k \geqslant n} x_k \leqslant M_0 < +\infty,$ 矛盾. 于是对于任意 $M > 0,$ 存在正整数 $N,$ 当 $n > N$ 时, 有 $x_n > M,$ 即 $\lim\limits_{n\to\infty} x_n = +\infty.$

\end{frame}

%------------------------------------------------

\begin{frame}{上、下极限的微观本质}

我们把之前证明中用到的类似的结论总结为如下定理, 以揭示上、下极限的微观本质:

\begin{thm}
设 $\{x_n\}$ 是有界数列, 那么
\begin{enumerate}
\item $\varlimsup\limits_{n\to\infty} x_n = H$ 的充要条件为 $\forall~\varepsilon > 0,$
\begin{enumerate}
\item 存在正整数 $N,$ 当 $n > N$ 时, 有 $x_n < H + \varepsilon;$
\item 存在无穷多个 $x_n > H - \varepsilon.$
\end{enumerate}
\item $\varliminf\limits_{n\to\infty} x_n = h$ 的充要条件为 $\forall~\varepsilon > 0,$
\begin{enumerate}
\item 存在正整数 $N,$ 当 $n > N$ 时, 有 $x_n > h - \varepsilon;$
\item 存在无穷多个 $x_n < h + \varepsilon.$
\end{enumerate}
\end{enumerate}
\end{thm}
\end{frame}

\section[运算]{上、下极限的运算}

%------------------------------------------------

\begin{frame}{上、下极限的运算性质}

一般来说, 极限运算具有的四则运算法则, 以及其他性质 (主要是等式), 对于上、下极限来说, \alert{都不成立. 但有一些不等式是成立的}.

\pause

\begin{thm}[加法运算性质]
设 $\{x_n\}, \{y_n\}$ 为数列, 则有
\begin{enumerate}
\item $\varlimsup\limits_{n\to\infty} (x_n + y_n) \leqslant \varlimsup\limits_{n\to\infty} x_n + \varlimsup\limits_{n\to\infty} y_n;$ $\varliminf\limits_{n\to\infty} (x_n + y_n) \geqslant \varliminf\limits_{n\to\infty} x_n + \varliminf\limits_{n\to\infty} y_n.$
\item 若 $\lim\limits_{n\to\infty} x_n$ 存在, 则有
\[
\varlimsup\limits_{n\to\infty} (x_n + y_n) = \lim\limits_{n\to\infty} x_n + \varlimsup\limits_{n\to\infty} y_n; \quad
\varliminf\limits_{n\to\infty} (x_n + y_n) = \lim\limits_{n\to\infty} x_n + \varliminf\limits_{n\to\infty} y_n.
\]
\end{enumerate}
\alert{注意:} 上面 4 个式子中, 若右端出现 $+\infty - \infty$ 的形式, 则该等式不成立.
\end{thm}

\pause

只证明第一个不等式, 另一个类似. 记 $H_x = \varlimsup\limits_{n\to\infty} x_n,$ $H_y = \varlimsup\limits_{n\to\infty} y_n.$ 可以不妨设 $H_x, H_y$ 均为有限值, 否则不等式显然成立. 那么 $\forall~\varepsilon > 0,$ 存在正整数 $N_1,$ 当 $n > N_1$ 时, 有 $x_n < H_x + \frac{\varepsilon}{2};$ 存在正整数 $N_2,$ 当 $n > N_2$ 时, 有 $y_n < H_y + \frac{\varepsilon}{2}.$ 取 $N = \max\{N_1, N_2\},$ 则当 $n > N$ 时, 有
\[
x_n + y_n < H_x + H_y + \varepsilon.
\]
因此 $\varlimsup\limits_{n\to\infty} (x_n + y_n) \leqslant H_x + H_y.$

\end{frame}

%------------------------------------------------

\begin{frame}{上、下极限的运算性质}

\begin{thm}[乘法运算性质]
设 $\{x_n\}, \{y_n\}$ 为数列, 则有
\begin{enumerate}
\item 若 $\{x_n\}, \{y_n\}$ 均为非负数列, 则有
\[\varlimsup\limits_{n\to\infty} (x_n y_n) \leqslant \varlimsup\limits_{n\to\infty} x_n \cdot \varlimsup\limits_{n\to\infty} y_n;\quad
\varliminf\limits_{n\to\infty} (x_n y_n) \geqslant \varliminf\limits_{n\to\infty} x_n \cdot \varliminf\limits_{n\to\infty} y_n.\]
\item 若 $\lim\limits_{n\to\infty} x_n = a,$ $0 < a < +\infty,$ 则有
\[
\varlimsup\limits_{n\to\infty} (x_n y_n) = a \cdot \varlimsup\limits_{n\to\infty} y_n;\quad
\varliminf\limits_{n\to\infty} (x_n y_n) = a \cdot \varliminf\limits_{n\to\infty} y_n.
\]
\end{enumerate}
\alert{注意:} 上面 4 个式子中, 若右端出现 $0 \cdot (+\infty)$ 的形式, 则该等式不成立.
\end{thm}

\pause

只证明第一个不等式, 另一个类似. 记 $H_x = \varlimsup\limits_{n\to\infty} x_n,$ $H_y = \varlimsup\limits_{n\to\infty} y_n.$ 类似地也可以不妨设 $H_x, H_y$ 均为有限值, 且大于 $0.$ 那么 $\forall~\varepsilon > 0,$ 存在正整数 $N_1,$ 当 $n > N_1$ 时, 有 $x_n < H_x + \frac{\varepsilon}{2H_y};$ 存在正整数 $N_2,$ 当 $n > N_2$ 时, 有 $y_n < H_y + \frac{\varepsilon}{2H_x}.$ 取 $N = \max\{N_1, N_2\},$ 则当 $n > N$ 时, 有
\[
x_n y_n < H_x H_y + \varepsilon.
\]
因此 $\varlimsup\limits_{n\to\infty} (x_n y_n) \leqslant H_x H_y.$
