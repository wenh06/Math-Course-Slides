\begin{frame}
\frametitle{绪言}

设 $f(x) = \sum_{n=0}^\infty a_n (x - x_0)^n$ 在 $x_0$ 的某个邻域 $B(x_0, \rho)$ 内成立, 那么由幂级数的逐项可导性定理, 可以归纳地得出, 在 $B(x_0, \rho)$ 内, $f$ 的任意可导, 且
\begin{align*}
f'(x) & = \sum_{n=1}^\infty n a_n (x - x_0)^{n-1}, \\
f''(x) & = \sum_{n=2}^\infty n (n-1) a_n (x - x_0)^{n-2}, \\
\vdots \\
f^{(k)}(x) & = \sum_{n=k}^\infty n (n-1) \cdots (n-k+1) a_n (x - x_0)^{n-k}.
\end{align*}
令 $x = x_0$, 可得 $f^{(k)}(x_0) = k! a_k$, 从而 $a_k = \frac{f^{(k)}(x_0)}{k!}$ (所谓 Taylor 系数).

\end{frame}

\section{Taylor 级数与余项公式}

%------------------------------------------------

\begin{frame}
\frametitle{Taylor 级数}

\begin{recall}[Taylor 公式]
设 $f$ 在 $x_0$ 的某个邻域内具有直到 $k$ 阶导数, 则在 $x_0$ 的某个邻域 $B_k(x_0, \rho_k)$ 内有
\[
f(x) = \sum_{k=0}^n \frac{f^{(k)}(x_0)}{k!} (x - x_0)^k + r_n(x),
\]
其中 $r_n(x)$ 是余项.
\end{recall}

由此, 可以引出 Taylor 级数的概念:

\pause
\vspace{0.5em}

\begin{Def}[Taylor 级数]
设 $f$ 在 $x_0$ 的某个邻域 $B(x_0, \rho)$ 内具有任意阶导数, 则称级数
\[
\sum_{n=0}^\infty \frac{f^{(n)}(x_0)}{n!} (x - x_0)^n
\]
为 $f$ 在 $x_0$ 的 Taylor 级数.
\end{Def}

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{Taylor 级数的收敛性}

\begin{question}
是否总存在正的常数 $\rho$ 使得在 $B(x_0, \rho)$ 内, $f$ 的 Taylor 级数收敛且等于 $f$, 即
\[
f(x) = \sum_{n=0}^\infty \frac{f^{(n)}(x_0)}{n!} (x - x_0)^n, \quad \forall x \in B(x_0, \rho)?
\]
\end{question}

\alert{注意}, 这样的函数称作是 \textbf{解析函数}.

\pause
\vspace{0.5em}

\alert{答案是否定的}. 在 Taylor 公式中
\[
r_k(x) = \frac{f^{(k+1)}(\xi)}{(k+1)!} (x - x_0)^{k+1}, \quad \xi \in B_k(x_0, \rho_k),
\]
余项 $r_k(x)$ 一般是用 $(x - x_0)^{k+1}$ 作为``尺''来``测量''的, 但有些无穷小量不能被任何一把这样的``尺''测量 (同阶无穷小), 从而导致 Taylor 级数无法收敛到 $f(x)$.

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{非解析函数的典型例子}

考察函数 ~ $f(x) = \begin{cases} e^{-1/x^2}, & x \neq 0, \\ 0, & x = 0. \end{cases}$

\pause
\vspace{0.5em}

可以算得
\[
f'(0) = \lim_{x \to 0} \frac{f(x) - f(0)}{x - 0} = \lim_{x \to 0} \frac{e^{-1/x^2}}{x} = \lim_{x \to 0} \frac{1/x}{e^{1/x^2}} = \lim_{x \to 0} \frac{-1/x^2}{(-2/x^3) e^{1/x^2}} = \lim_{x \to 0} \frac{x}{2 e^{1/x^2}} = 0.
\]
当 $x \neq 0$ 时, $f'(x) = \frac{2}{x^3} e^{-1/x^2} = P_3 \left(\frac{1}{x}\right) e^{-1/x^2}$, 其中 $P_3$ 是一个三次多项式.

\pause
\vspace{0.5em}

接下来, 可以计算 2 阶导数:
\[
f''(0) = \lim_{x \to 0} \frac{f'(x) - f'(0)}{x - 0} = \lim_{x \to 0} \frac{P_3(1/x) e^{-1/x^2}}{x} = \lim_{x \to 0} \frac{2 / x^4}{e^{1/x^2}} = 0.
\]
当 $x \neq 0$ 时, $f''(x) = -\frac{6}{x^4} e^{-1/x^2} + \frac{2}{x^3} \cdot \frac{2}{x^3} e^{-1/x^2} = \left(-\frac{6}{x^4} + \frac{4}{x^6}\right) e^{-1/x^2} = P_6 \left(\frac{1}{x}\right) e^{-1/x^2}$, 其中 $P_6$ 是一个六次多项式.

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{非解析函数的典型例子}

\begin{empheq}[box={\fancymath[colback=cyan!30, drop lifted shadow, sharp corners]}]{equation}
\text{可以猜测} \quad f^{(n)}(x) = \begin{cases} P_{3n} \left(\frac{1}{x}\right) e^{-1/x^2}, & x \neq 0, \\ 0, & x = 0. \end{cases}
\tag{$\star$}
\label{eq:star}
\end{empheq}

\pause
\vspace{0.5em}

接下来用数学归纳法证明上述猜测. 我们已经验证了 $n = 0, 1, 2$ 的情况. 对于 $n + 1:$
\[
f^{(n+1)}(0) = \lim_{x \to 0} \frac{f^{(n)}(x) - f^{(n)}(0)}{x - 0} = \lim_{x \to 0} \frac{P_{3n}(1/x) e^{-1/x^2}}{x} = \lim_{x \to 0} \frac{P_{3n+1}(1/x)}{e^{1/x^2}} = 0,
\]
\pause
当 $x \neq 0$ 时,
\begin{align*}
f^{(n+1)}(x) & = P_{3n}'\left(\frac{1}{x}\right) \cdot \left(\frac{1}{x}\right)' \cdot e^{-1/x^2} + P_{3n}\left(\frac{1}{x}\right) \cdot \left(e^{-1/x^2}\right)' \\
& = \left(-\frac{1}{x^2} P_{3n}'\left(\frac{1}{x}\right) + \frac{2}{x^3} P_{3n}\left(\frac{1}{x}\right)\right) e^{-1/x^2} = P_{3n+3}\left(\frac{1}{x}\right) e^{-1/x^2}.
\end{align*}

\pause

于是, 我们证明了 $f(x)$ 在 $x = 0$ 处的 Taylor 级数为 $\sum_{n=0}^\infty \frac{f^{(n)}(0)}{n!} x^n = 0$, 但 $f(x)$ 显然不是常值函数 $0.$ 也就是说, $f(x)$ 不是解析函数.
\qed

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{$C^\infty$ 与 $C^\omega$: 光滑函数与解析函数}

以上例子说明, 无穷次可微未必解析. 记
\begin{itemize}
\item $C^\infty(\mathbb{R})$: \textbf{光滑函数}全体, 即在 $\mathbb{R}$ 上具有任意阶导数的函数全体;
\item $C^\omega(\mathbb{R})$: \textbf{实解析函数}全体, 即在每点附近都能展开为收敛幂级数的函数全体.
\end{itemize}
显然 $C^\omega(\mathbb{R}) \subseteq C^\infty(\mathbb{R})$, 且上面的例子说明这一包含关系\alert{严格}: $C^\omega(\mathbb{R}) \subsetneqq C^\infty(\mathbb{R})$.

\pause
\vspace{0.5em}

\begin{question}
$C^\infty$ 函数的各阶导数值在多大程度上是``自由''的? 任给一列实数 $a_0, a_1, a_2, \ldots$, 能否构造 $f \in C^\infty(\mathbb{R})$ 使得 $f^{(k)}(0) = a_k$ 对所有 $k \geqslant 0$ 成立?
\end{question}

\pause
\vspace{0.5em}

若 $f \in C^\omega(\mathbb{R})$, 则 $f$ 由其在 $0$ 处的各阶导数值完全确定 (Taylor 系数 $a_k = \frac{f^{(k)}(0)}{k!}$), 且 $\sum \frac{a_k}{k!} x^k$ 必须在某个邻域内收敛. 因此对解析函数, $(a_k)$ \alert{不能任意给定}.

\pause
\vspace{0.5em}

对 $C^\infty$ 函数, \alert{答案是肯定的}, 即 $(a_k)$ 可以任意给定, 这正是 \textbf{Borel 引理}的内容.

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{截断函数 (Bump Function) 的显式构造}

构造 Borel 引理的关键工具是\textbf{截断函数}. 可以构造这么一个例子, 分三步进行:

\pause
\vspace{0.3em}

\textbf{第一步}: 令 $\alpha(x) = \begin{cases} e^{-1/x^2}, & x > 0, \\ 0, & x \leqslant 0. \end{cases}$ 由前面的论证, $\alpha \in C^\infty(\mathbb{R})$, 且 $\alpha > 0$ 当且仅当 $x > 0$.

\pause
\vspace{0.3em}

\textbf{第二步}: 令 $\beta(x) = \dfrac{\alpha(x)}{\alpha(x) + \alpha(1-x)}.$ 由于对一切 $x \in \mathbb{R}$, 分母 $\alpha(x) + \alpha(1-x) > 0$, 故 $\beta \in C^\infty(\mathbb{R})$, 且
\[
\beta(x) = \begin{cases} 0, & x \leqslant 0, \\ \in (0,1), & 0 < x < 1, \\ 1, & x \geqslant 1. \end{cases}
\]

\pause
\vspace{0.3em}

\textbf{第三步}: 令 $\varphi(x) = \beta(x + 2) \cdot \beta(2 - x).$ 则 $\varphi \in C^\infty(\mathbb{R})$, 且
\begin{empheq}[box={\fancymath[colback=cyan!30, drop lifted shadow, sharp corners]}]{equation*}
\varphi(x) = \begin{cases} 1, & |x| \leqslant 1, \\ \in (0,1), & 1 < |x| < 2, \\ 0, & |x| \geqslant 2. \end{cases}
\end{empheq}

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{Borel 引理}

\begin{prop}[Borel 引理]
设 $(a_k)_{k \geqslant 0}$ 是任意一列实数, 则存在 $f \in C^\infty(\mathbb{R})$, 使得
\[
    f^{(k)}(0) = a_k, \quad k = 0, 1, 2, \ldots
\]
\end{prop}

\pause
\vspace{0.5em}

\startproof[bg=yellow, fg=red](证明思路)
设 $\varphi$ 是上述截断函数. 令
\[
f(x) = \sum_{k=0}^\infty \frac{a_k}{k!} \, \varphi(\lambda_k x) \, x^k,
\]
其中正数 $\lambda_k$ 待定. 由于 $\varphi(\lambda_k x) = 0$ 当 $|x| \geqslant \frac{2}{\lambda_k}$, 故选取 $\lambda_k$ 足够大, 可使第 $k$ 项在 $C^{k-1}$ 意义下 (指的是 $(k-1)$-阶导数绝对值上确界) 的范数小于 $2^{-k}$, 从而保证级数在 $C^\infty$ 意义下一致收敛.

\pause
\vspace{0.5em}

验证导数值: 对每个固定的 $k$, 当 $x$ 充分靠近 $0$ 时, $\varphi(\lambda_j x) = 1$ 对所有 $j \leqslant k$ 成立, 而 $j > k$ 的各项在 $x = 0$ 处贡献为 $0$ (因子 $x^j$ 使 $j$ 阶以内的导数为零). 逐项求导后令 $x = 0$, 可得 $f^{(k)}(0) = a_k$. \qed

\end{frame}

\begin{frame}
\frametitle{计算和函数的反问题}

\begin{question}
什么条件下, 存在正的常数 $\rho$ 使得在 $B(x_0, \rho)$ 内, $f$ 的 Taylor 级数收敛且等于 $f$, 即
\[
f(x) = \sum_{n=0}^\infty \frac{f^{(n)}(x_0)}{n!} (x - x_0)^n, \quad \forall x \in B(x_0, \rho)?
\]
\end{question}

\pause
\vspace{0.5em}

还是从 Taylor 公式出发
\[
f(x) = \sum_{k=0}^n \frac{f^{(k)}(x_0)}{k!} (x - x_0)^k + r_n(x),
\]
那么 $f(x)$ 在 $B(x_0, \rho)$ 内的 Taylor 级数收敛且等于 $f(x)$ 的\alert{充分必要条件}是
\begin{empheq}[box={\fancymath[colback=cyan!30, drop lifted shadow, sharp corners]}]{equation*}
\lim_{n \to \infty} r_n(x) = 0, \quad \forall x \in B(x_0, \rho). \quad (\text{\color{red}不要求一致性})
\end{empheq}

\pause
\vspace{0.5em}

之前的例子 $f(x) = \begin{cases} e^{-1/x^2}, & x \neq 0, \\ 0, & x = 0. \end{cases}$ 中, $r_n(x) \equiv e^{-1/x^2}$ (\alert{不依赖于 $n$}), 从而 $\lim_{n \to \infty} r_n(x) = e^{-1/x^2} \neq 0.$

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{积分形式的余项公式}

于是, 研究一个函数 $f$ 是否是解析函数, 或者说它是否存在幂级数展开 (Taylor 展开), 就转化为了研究 Taylor 公式中的余项 $r_n(x)$ 的极限行为.

\begin{recall}[各种形式的余项公式]
\begin{itemize}
\item Peano: $r_n(x) = o((x - x_0)^n)$
\item Lagrange: $r_n(x) = \frac{f^{(n+1)}(\xi)}{(n+1)!} (x - x_0)^{n+1} = \frac{f^{(n+1)}(x_0 + \theta (x - x_0))}{(n+1)!} (x - x_0)^{n+1}$
\item Cauchy: $r_n(x) = \frac{f^{(n+1)}(c)}{n!} (x - c)^n (x - x_0) = \frac{f^{(n+1)}(x_0 + \theta (x - x_0))}{n!} (1 - \theta)^n (x - x_0)^{n+1}$
\end{itemize}
\end{recall}

\pause
\vspace{0.5em}

\begin{Def}[积分形式的余项公式]
设 $f(x)$ 在 $x_0$ 的某个邻域 $B(x_0, \rho)$ 内具有 $n + 1$ 阶导数, 则在 $B(x_0, \rho)$ 内有
\[
f(x) = \sum_{k=0}^n \frac{f^{(k)}(x_0)}{k!} (x - x_0)^k + {\color{red} \frac{1}{n!} \int_{x_0}^x f^{(n+1)}(t) (x - t)^n ~ \mathrm{d}t}.
\]
\end{Def}

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{积分形式余项公式的证明}

由定义, 余项 $r_n(x) = f(x) - \sum_{k=0}^n \frac{f^{(k)}(x_0)}{k!} (x - x_0)^k,$ 那么 $\forall ~ k \leqslant n,$
\[
r_n^{(k)}(x) = f^{(k)}(x) - \sum_{j=k}^n \frac{f^{(j)}(x_0)}{j!} j (j-1) \cdots (j-k+1) (x - x_0)^{j-k},
\]
从而 $r_n^{(k)}(x_0) = f^{(k)}(x_0) - \frac{f^{(k)}(x_0)}{k!} k (k-1) \cdots 1 = 0.$

\pause
\vspace{0.5em}

再反过来逐次应用分部积分, 可以得到
\begin{align*}
r_n(x) & = \int_{x_0}^x r_n'(t) ~ \mathrm{d}t = \int_{x_0}^x r_n'(t) ~ {\color{red} \mathrm{d} (t-x)} = \int_{x_0}^x r_n''(t) \cdot (x - t) ~ \mathrm{d}t = - \int_{x_0}^x r_n''(t) ~ \mathrm{d} \frac{(x - t)^2}{2} \\
& = \cdots = \left( \int_{x_0}^x r_n^{(k)}(t) \cdot (x - t)^{k-1} ~ \mathrm{d} t \right) \cdot \frac{1}{(k-1)!} = \cdots = \frac{1}{n!} \int_{x_0}^x {\color{red} r_n^{(n+1)}(t)} (x - t)^n ~ \mathrm{d}t \\
& = \frac{1}{n!} \int_{x_0}^x {\color{red} f^{(n+1)}(t)} (x - t)^n ~ \mathrm{d}t.
\end{align*}

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{积分形式余项公式}

\begin{remark}
积分形式的余项公式可以 ({\color{red}利用积分第一中值定理}) 推出其它形式的余项公式:
\begin{itemize}
\item Lagrange 形式:
\begin{align*}
r_n(x) & = \frac{1}{n!} \int_{x_0}^x ~ {\color{red} f^{(n+1)}(t)} \underbrace{{\color{cyan}(x - t)^n}}_{\text{不变号}} ~ \mathrm{d}t = \frac{1}{n!} ~ {\color{red} f^{(n+1)}(\xi)} \int_{x_0}^x ~ {\color{cyan} (x - t)^n} ~ \mathrm{d}t \\
& = \frac{f^{(n+1)}(\xi)}{(n+1)!} \left. \left( -(x-t)^{n+1} \right) \right|_{t=x_0}^{t=x} = \frac{f^{(n+1)}(\xi)}{(n+1)!} (x - x_0)^{n+1}.
\end{align*}
\pause
\item Cauchy 形式:
\begin{align*}
r_n(x) & = \frac{1}{n!} \int_{x_0}^x ~ {\color{red} f^{(n+1)}(t) (x - t)^n} ~ \mathrm{d}t = \frac{1}{n!} ~ {\color{red} f^{(n+1)}(c) (x - c)^n} \int_{x_0}^x ~ \mathrm{d}t \\
& = \frac{f^{(n+1)}(c)}{n!} (x - c)^n (x - x_0).
\end{align*}
\end{itemize}
\end{remark}

\end{frame}

\section{初等函数的幂级数展开}

%------------------------------------------------

\begin{frame}
\frametitle{指数函数的幂级数展开}

{\larger[1]{\color{red} \ding{43}} 计算 $e^x$ 在 $x = 0$ 处的幂级数展开.}

\pause
\vspace{0.5em}

之前计算函数项级数 $\sum_{n=0}^\infty \frac{x^n}{n!}$ 的和函数时, 已经验证了 $e^x = \sum_{n=0}^\infty \frac{x^n}{n!}$ 在 $\mathbb{R}$ 内成立. 现在, 从 Taylor 公式出发, 验证其余项 $r_n(x) \to 0$ 的极限行为:

\pause
\vspace{0.5em}

对于 $x \in \mathbb{R}$ 固定, 余项
\begin{itemize}
\item Lagrange 形式: $r_n(x) = \frac{e^\xi}{(n+1)!} x^{n+1} \xrightarrow{n \to \infty} 0$;
\item Cauchy 形式: $r_n(x) = \frac{e^c}{n!} (x - c)^n x \xrightarrow{n \to \infty} 0$.
\end{itemize}
所以 $e^x$ 在 $x = 0$ 处的幂级数展开为 $\sum_{n=0}^\infty \frac{x^n}{n!}$.
\qed

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{三角函数的幂级数展开 | $\sin x, \cos x$}

{\larger[1]{\color{red} \ding{43}} 计算 $\sin x, \cos x, \dots$ 在 $x = 0$ 处的幂级数展开.}

\pause
\vspace{0.5em}

由于 $\sin^{(n)}(x) = \sin (x + n\pi/2),$ 从而 $\sin^{(n)}(0) = \begin{cases} 0, & n = 2k \text{ 是偶数}, \\ (-1)^{\frac{n-1}{2}}, & n = 2k+1 \text{ 是奇数}. \end{cases}$ 所以 $\sin x$ 在 $x = 0$ 处的幂级数为
\[
\sum_{k=0}^n \frac{\sin^{(k)}(0)}{k!} x^k + r_n(x) = \sum_{k=0}^n \frac{(-1)^k}{(2k+1)!} x^{2k+1} + r_{2n + 2}(x),
\]
其中 $r_{2n + 2}(x) = \frac{1}{(2n+3)!} \underbrace{\sin^{(2n+3)}(\xi)}_{\text{绝对值} \leqslant 1} x^{2n+3} \xrightarrow{n \to \infty} 0.$ 所以 $\sin x$ 在 $x = 0$ 处的幂级数展开为
\[
\sin x = \sum_{k=0}^\infty \frac{(-1)^k}{(2k+1)!} x^{2k+1}, ~~ \text{收敛半径为 } r = 1 / \varlimsup_{k \to \infty} \nroot[2k+1]{\frac{1}{(2k+1)!}} = +\infty.
\]

\pause
\vspace{0.5em}

类似可求得 $\cos x$ 在 $x = 0$ 处的幂级数展开为
\[
\cos x = \sum_{k=0}^\infty \frac{(-1)^k}{(2k)!} x^{2k}, ~~ \text{收敛半径为 } r = +\infty.
\]

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{三角函数的幂级数展开 | $\tan x$}

$f(x) = \tan x$ 的 Taylor 公式为 $f(x) = \sum_{k=0}^n \frac{f^{(k)}(0)}{k!} x^k + r_n(x).$
\begin{align*}
f'(x) & = \cos^{-2} x \quad \implies f'(0) = 1, \\
f''(x) & = 2 \cos^{-3} x \sin x \quad \implies f''(0) = 0, \\
f'''(x) & = -6 \cos^{-4} x \sin^2 x + 2 \cos^{-3} x \cos x \quad \implies f'''(0) = 2 \cdot 3!, \\
& \hspace{1em} \vdots \\
\tan x & = x + \frac{2}{3!} x^3 + (\text{后面系数的直接计算比较麻烦}).
\end{align*}
需要另想办法计算 $\tan x$ 的幂级数展开.

\pause
\vspace{0.5em}

注意到 $\tan x = \frac{\sin x}{\cos x}$, 于是有 $\sin x = \tan x \cdot \cos x$, 可以通过待定系数法来计算 $\tan x$ 的幂级数展开. 由于 $\tan x$ 为奇函数, 可设 $\tan x = c_1 x + c_3 x^3 + c_5 x^5 + \cdots$, 那么
\[
\sum_{n=0}^\infty \frac{(-1)^n}{(2n+1)!} x^{2n+1} = \sin x = \tan x \cdot \cos x = \left( \sum_{n=0}^\infty c_{2n+1} x^{2n+1} \right) \left( \sum_{n=0}^\infty \frac{(-1)^n}{(2n)!} x^{2n} \right).
\]

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{三角函数的幂级数展开 | $\tan x$}

比较上式两端的系数, 可以得到

n = 0: & \frac{1}{1!} = c_1 \cdot \frac{1}{0!} \implies c_1 = 1; \\

n = 2: & \frac{1}{5!} = c_5 \cdot \frac{1}{0!} + c_3 \cdot \frac{-1}{2!} + c_1 \cdot \frac{1}{4!} \implies c_5 = \frac{16}{5!}; \\

\begin{frame}
\frametitle{三角函数的幂级数展开 | $\tan x$}

我们通过其他方法来计算 $\tan x$ 的幂级数展开的收敛半径. 由于
\begin{align*}
\tan x & = \frac{\sin x}{\cos x} = \left. \frac{e^{ix} - e^{-ix}}{2i} \middle/ \frac{e^{ix} + e^{-ix}}{2} \right. = \frac{e^{2ix} - 1}{i (e^{2ix} + 1)} = i \cdot \frac{e^{-2ix} - 1}{e^{-2ix} + 1}, \\
\cot x & = \frac{\cos x}{\sin x} = i \cdot \frac{e^{2ix} + 1}{e^{2ix} - 1} = i \cdot \left( \frac{2}{e^{2ix} - 1} + 1 \right),
\end{align*}
所以有 $x \cot x = ix + \frac{2ix}{e^{2ix} - 1} \xlongequal{z = 2ix} \frac{z}{2} + \frac{z}{e^z - 1}.$ 这里的 $\frac{z}{e^z - 1}$ 是所谓的 Bernoulli 数列的生成函数:
\pause
\begin{empheq}[box={\fancymath[colback=cyan!30, drop lifted shadow, sharp corners]}]{equation*}
\frac{z}{e^z - 1} = \sum_{n=0}^\infty \frac{B_n}{n!} z^n, \quad \text{其中 } B_n \text{ 是第 $n$ 个 Bernoulli 数}.
\end{empheq}

\pause
\vspace{0.5em}

接下来我们需要讨论 Bernoulli 数列的计算及其性质, 从而方便计算 $\tan x$ 的幂级数展开的收敛半径.

\end{frame}

\begin{frame}
\frametitle{Bernoulli 数}

Bernoulli 数可以利用待定系数法, 通过递推的方式计算:
\[
z = (e^z - 1) \sum_{n=0}^\infty \frac{B_n}{n!} z^n = \left( \sum_{n=1}^\infty \frac{z^n}{n!} \right) \left( \sum_{n=0}^\infty \frac{B_n}{n!} z^n \right).
\]
上式左边 $z^m (m \geqslant 1)$ 的系数为 $\begin{cases} 1, & m = 1, \\ 0, & m > 1. \end{cases}$ 上式右边 $z^m$ 的系数为
\[
\frac{B_0}{0!} \cdot \frac{1}{m!} + \frac{B_1}{1!} \cdot \frac{1}{(m-1)!} + \cdots + \frac{B_{m-1}}{(m-1)!} \cdot \frac{1}{1!} = \frac{1}{m!} \sum_{n=0}^{m-1} C_m^n B_n.
\]
比较上式两端的系数, 可以得到 $\sum_{n=0}^{m-1} C_m^n B_n = \begin{cases} 1, & m = 1, \\ 0, & m > 1. \end{cases}$
\pause
由此可推出 $B_m$ 的一般计算公式:
\begin{empheq}[box={\fancymath[colback=cyan!30, drop lifted shadow, sharp corners]}]{equation*}
B_m = \sum_{n=0}^{m} \frac{1}{n+1} \sum_{k=0}^n (-1)^k C_n^k k^m.
\end{empheq}

\end{frame}

\begin{frame}
\frametitle{Bernoulli 数的性质}

Bernoulli 数列的前几个数值为
\[
B_0 = 1, \quad B_1 = -\frac{1}{2}, \quad B_2 = \frac{1}{6}, \quad B_3 = 0, \cdots
\]

\pause
\vspace{0.5em}

\begin{prop}[Bernoulli 数的性质]
\begin{itemize}
\item $B_{2n+1} = 0$ 对所有 $n \geqslant 1$ 成立;
\item $B_{2n} = (-1)^{n+1} \cdot \frac{2 (2n)!}{(2\pi)^{2n}} \cdot \zeta(2n)$ 对所有 $n \geqslant 1$ 成立.
\end{itemize}
\end{prop}

\pause
\vspace{0.5em}

\startproof[bg=yellow, fg=red]
{\color{red} \ding{43}} 第 1 点关于 Bernoulli 数列的奇数项为零的证明相对简单. 考虑
\[
f(z) = \frac{z}{e^z - 1} + \frac{z}{2} = 1 - \frac{1}{2} z + \sum_{m=2}^\infty \frac{B_m}{m!} z^m + \frac{z}{2} = 1 + \sum_{m=2}^\infty \frac{B_m}{m!} z^m.
\]
那么
\[
f(-z) = \frac{-z}{e^{-z} - 1} + \frac{-z}{2} = \frac{z e^z}{e^z - 1} - \frac{z}{2} = \frac{z e^z - z + z}{e^z - 1} - \frac{z}{2} = \frac{z}{e^z - 1} + \frac{z}{2} = f(z).
\]

\end{frame}

\begin{frame}
\frametitle{Bernoulli 数的性质}

所以 $f(z) = 1 + \sum_{m=2}^\infty \frac{B_m}{m!} z^m$ 是偶函数, 从而 $B_{2n+1} = 0$ 对所有 $n \geqslant 1$ 成立.

\pause
\vspace{0.5em}

{\color{red} \ding{43}} 接下来证明 Bernoulli 数列的偶数项 $B_{2n}$ 的计算公式. 将 $\frac{z}{e^z - 1} = 1 - \frac{z}{2} + \sum_{n=1}^\infty \frac{B_{2n}}{(2n)!} z^{2n}$ 代入 $x \cot x = ix + \frac{2ix}{e^{2ix} - 1}$ 中, 可以得到
\[
x \cot x = ix + \left( 1 - \frac{2ix}{2} + \sum_{n=1}^\infty \frac{B_{2n}}{(2n)!} (2ix)^{2n} \right) = 1 + \sum_{n=1}^\infty (-1)^n \cdot \frac{2^{2n} B_{2n}}{(2n)!} x^{2n}.
\]
从而可得 $\cot x = \frac{1}{x} + \sum_{n=1}^\infty (-1)^n \cdot \frac{2^{2n} B_{2n}}{(2n)!} x^{2n-1}.$

\pause
\vspace{0.5em}

另一方面, 对 $\sin x$ 的无穷乘积展开 $\frac{\sin \pi x}{\pi x} = \prod_{n=1}^\infty \left( 1 - \frac{x^2}{n^2} \right)$ 两端取对数导数 $\frac{1}{\pi} \cdot \frac{\mathrm{d}}{\mathrm{d}x} \ln ( \cdot ),$ 可得
\[
\frac{\cos \pi x}{\sin \pi x} - \frac{1}{\pi x} = - \frac{1}{\pi} \sum_{n=1}^\infty \frac{2x}{n^2} \cdot \frac{1}{1 - \frac{x^2}{n^2}},
\]

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{Bernoulli 数的性质}

从而有 (以下设 $|x| < 1$)
\begin{align*}
\pi x \cot \pi x & = 1 - 2 \sum_{n=1}^\infty \frac{x^2}{n^2} \cdot \frac{1}{1 - \frac{x^2}{n^2}} = 1 - 2 \sum_{n=1}^\infty \frac{x^2}{n^2} \cdot \left( \sum_{k=0}^\infty \left( \frac{x^2}{n^2} \right)^k \right) = 1 - 2 ~ {\color{red} \sum_{n=1}^\infty \sum_{k=1}^\infty} ~ \left( \frac{x^2}{n^2} \right)^k \\
& = 1 - 2 ~ {\color{red}\sum_{k=1}^\infty} ~ \left( {\color{red}\sum_{n=1}^\infty} ~ \frac{1}{n^{2k}} \right) x^{2k} = 1 - 2 \sum_{k=1}^\infty \zeta(2k) x^{2k}.
\end{align*}
\pause
与之前算得的 $x \cot x = 1 + \sum_{n=1}^\infty (-1)^n \cdot \frac{2^{2n} B_{2n}}{(2n)!} x^{2n}$ 相比较, 可得
\[
-2 \zeta(2n) = (-1)^n \cdot \frac{2^{2n} B_{2n}}{(2n)!} \pi^{2n} ~~ \implies ~~ B_{2n} = (-1)^{n+1} \cdot \frac{2 (2n)!}{(2\pi)^{2n}} \cdot \zeta(2n).
\]
\qed

\end{frame}

\begin{frame}
\frametitle{三角函数的幂级数展开 | $\tan x$}

我们回到 $\tan x$ 的幂级数展开. 之前算得 $\cot x = \frac{1}{x} + \sum_{n=1}^\infty (-1)^n \cdot \frac{2^{2n} B_{2n}}{(2n)!} x^{2n-1}.$ 如果利用关系式 $\tan x = \frac{1}{\cot x}$ 来计算 $\tan x$ 的幂级数展开, 仍然是比较麻烦的. 需要另想办法.

\pause
\vspace{0.5em}

\alert{可以利用 2 倍角相关的公式}:
\[
2 \cot 2x = \frac{\cos^2 x - \sin^2 x}{\sin x \cos x} = \cot x - \tan x ~~ \implies ~~ \tan x = \cot x - 2 \cot 2x.
\]
所以 $\tan x$ 的幂级数展开为
\begin{align*}
\tan x & = \cot x - 2 \cot 2x \\
& = \left( \frac{1}{x} + \sum_{n=1}^\infty (-1)^n \cdot \frac{2^{2n} B_{2n}}{(2n)!} x^{2n-1} \right) - 2 \left( \frac{1}{2x} + \sum_{n=1}^\infty (-1)^n \cdot \frac{2^{2n} B_{2n}}{(2n)!} (2x)^{2n-1} \right) \\
& = \sum_{n=1}^\infty (-1)^n \cdot \frac{4^n (1 - 4^n) B_{2n}}{(2n)!} x^{2n-1}.
\end{align*}

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{三角函数的幂级数展开 | $\tan x$}

设 $\tan x = \sum_{n=1}^\infty (-1)^n \cdot \frac{4^n (1 - 4^n) B_{2n}}{(2n)!} x^{2n-1}$ 的收敛半径为 $r$, 那么
\begin{align*}
\frac{1}{r^2} & = \lim_{n \to \infty} \left| \frac{4^{n+1} (1 - 4^{n+1}) B_{2n+2}}{(2n+2)!} \middle/ \frac{4^n (1 - 4^n) B_{2n}}{(2n)!} \right| = \lim_{n \to \infty} \left| \frac{2 (4^{n+1} - 1) B_{2n+2}}{(2n + 1)(n + 1) (4^n - 1) B_{2n}} \right| \\
& = \lim_{n \to \infty} \left| \frac{2 (4^{n+1} - 1) \cdot (-1)^{n+2} \cdot \frac{2 (2n+2)!}{(2\pi)^{2n+2}} \cdot \zeta(2n+2)}{(2n + 1)(n + 1) (4^n - 1) \cdot (-1)^{n+1} \cdot \frac{2 (2n)!}{(2\pi)^{2n}} \cdot \zeta(2n)} \right| \\
& = \lim_{n \to \infty} \left| 2 \cdot \frac{4^{n+1} - 1}{4^n - 1} \cdot \frac{(2n+2)(2n+1)}{(2n + 1)(n + 1)} \cdot \frac{\zeta(2n+2)}{\zeta(2n)} \cdot \frac{1}{4\pi^2} \right| \\
& = 2 \cdot 4 \cdot 2 \cdot 1 \cdot \frac{1}{4\pi^2} = \frac{4}{\pi^2} ~~ \implies ~~ r = \frac{\pi}{2}.
\end{align*}
注意, 这里用了 $\lim_{s \to +\infty} \zeta(s) = \lim_{s \to +\infty} \sum_{n=1}^\infty \frac{1}{n^s} = 1$ 的性质.

\pause
\vspace{0.5em}

所以 $\tan x$ 的幂级数展开的收敛半径为 $r = \frac{\pi}{2}$, 与 $\tan x$ 在 $x = \pm \frac{\pi}{2}$ 处的极点相符.
\qed

\end{frame}

\begin{frame}
\frametitle{三角函数的幂级数展开 | $\tan x$ 解析性的说明}

我们实际上是用``待定系数''的方法, 求得了一个幂级数 $T(x) = \sum_{n=1}^\infty c_{2n-1} x^{2n-1} = \sum_{n=1}^\infty (-1)^n \cdot \frac{4^n (1 - 4^n) B_{2n}}{(2n)!} x^{2n-1}$ 满足
\begin{itemize}
\item $T(x)$ 的收敛半径为 $r = \frac{\pi}{2}$;
\item $T(x)$ 满足 $T(x) \cdot \cos x = \sin x$.
\end{itemize}
所以在 $|x| < \frac{\pi}{2}$ 内, $T(x) = \frac{\sin x}{\cos x} = \tan x.$ 由幂级数展开的唯一性, 可得 $\tan x$ 的幂级数展开为
\[
\tan x = \sum_{n=1}^\infty (-1)^n \cdot \frac{4^n (1 - 4^n) B_{2n}}{(2n)!} x^{2n-1}, ~~ \text{收敛半径为 } r = \frac{\pi}{2}.
\]

\pause
\vspace{0.5em}

\alert{注意}, 用 Taylor 公式余项 $r_n(x) = \frac{f^{(n+1)}(\xi)}{(n+1)!} x^{n+1}$ 的极限行为来验证 $\tan x$ 的幂级数展开的正确性是比较麻烦的, 因为 $\tan x$ 的 $n$ 阶导数的表达式比较复杂, 从而 $r_n(x)$ 的表达式也比较复杂. 但是通过上面的方法, 可以比较简洁地验证 $\tan x$ 的幂级数展开的正确性.

\end{frame}

\begin{frame}
\frametitle{反三角函数的幂级数展开}

由于
\begin{gather*}
(\arcsin x)' = \frac{1}{\sqrt{1 - x^2}} = (1 - x^2)^{-\frac{1}{2}}, \quad (\arccos x)' = - \frac{1}{\sqrt{1 - x^2}} = - (1 - x^2)^{-\frac{1}{2}}, \\
(\arctan x)' = \frac{1}{1 + x^2} = (1 + x^2)^{-1}, \quad (\arccot x)' = - \frac{1}{1 + x^2} = - (1 + x^2)^{-1},
\end{gather*}
由逐项积分定理, 求反三角函数的幂级数展开, 就转化为了求幂函数
\[
f(x) = (1 + x)^{\alpha}
\]
的幂级数展开.

\end{frame}

\begin{frame}
\frametitle{幂函数 $f(x) = (1 + x)^{\alpha}$ 的幂级数展开}

{\color{red} \ding{43}} 首先考虑最简单的情况: $\alpha = m$ 是非负整数, 那么 $\forall ~ n > m,$ $f^{(n)}(0) = 0,$ 从而
\[
f(x) = \sum_{k=0}^m C_m^k x^k, \quad \text{收敛半径} ~ r = +\infty.
\]
``幂级数''即为二项展开所得的多项式.

\pause
\vspace{0.5em}

{\color{red} \ding{43}} 当 $\alpha$ 不是非负整数时, $\forall n \in \mathbb{N},$
\[
f^{(n)}(x) = \alpha (\alpha - 1) \cdots (\alpha - n + 1) (1 + x)^{\alpha - n} = C_\alpha^n n! (1 + x)^{\alpha - n},
\]
所以幂级数展开为 $\sum_{n=0}^\infty C_\alpha^n x^n.$ 其收敛半径可由 d'Alembert 定理计算
\[
\frac{1}{r} = \lim_{n \to \infty} \left| \frac{C_\alpha^{n+1}}{C_\alpha^n} \right| = \lim_{n \to \infty} \left| \frac{\alpha - n}{n + 1} \right| = 1 ~~ \implies ~~ r = 1.
\]
余项 ($|x| < 1$) 有
\begin{itemize}
\item Lagrange 形式:
\[
r_n(x) = \frac{f^{(n+1)}(\xi)}{(n+1)!} x^{n+1} = C_\alpha^{n+1} (1 + \xi)^{\alpha - n - 1} x^{n+1} = (1 + \xi)^{\alpha} \underbrace{C_\alpha^{n+1} x^{n+1}}_{\text{收敛级数通项} \to 0} \underbrace{\left( \frac{1}{1 + \xi} \right)^{n+1}}_{\text{$x < 0$ 时不好判断}}.
\]
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{幂函数 $f(x) = (1 + x)^{\alpha}$ 的幂级数展开}

\begin{itemize}
\item Cauchy 形式:
\begin{align*}
r_n(x) & = \frac{f^{(n+1)}(c)}{n!} (x - c)^n x = (n + 1) C_\alpha^{n+1} (1 + c)^{\alpha - n - 1} (x - c)^n x \\
& = (n + 1) C_\alpha^{n+1} (1 + \theta x)^{\alpha - n - 1} \cdot (1 - \theta)^n x^{n+1} \\
& = (1 + \theta x)^{\alpha-1} \cdot \underbrace{(n + 1) C_\alpha^{n+1} x^{n+1}}_{\text{收敛级数通项} \to 0} \cdot \left( \underbrace{\frac{1 - \theta}{1 + \theta x}}_{\in (0, 1)} \right)^n \to 0.
\end{align*}
\end{itemize}
因此在 $|x| < 1$ 内, $f(x) = (1 + x)^{\alpha}$ 的幂级数展开为 $\sum_{n=0}^\infty C_\alpha^n x^n.$

\pause
\vspace{0.5em}

{\color{red} \ding{43}} 接下来讨论收敛域边界 $x = \pm 1$ 上的情况.
\pause
对幂级数系数 $C_\alpha^n$ 做一个变形
{\smaller[1]
\begin{align*}
C_\alpha^n & = \frac{\alpha (\alpha - 1) \cdots (\alpha - n + 1)}{n!} = \frac{\alpha}{1} \cdot \frac{\alpha - 1}{2} \cdots \frac{\alpha - n + 1}{n} = \frac{\alpha+1-1}{1} \cdot \frac{\alpha+1-2}{2} \cdots \frac{\alpha+1-n}{n} \\
& = \left( -1 + \frac{\alpha+1}{1} \right) \cdot \left( -1 + \frac{\alpha+1}{2} \right) \cdots \left( -1 + \frac{\alpha+1}{n} \right) = (-1)^n \left( 1 - \frac{\alpha+1}{1} \right) \cdot \left( 1 - \frac{\alpha+1}{2} \right) \cdots \left( 1 - \frac{\alpha+1}{n} \right).
\end{align*}
}

\end{frame}

\begin{frame}
\frametitle{幂函数 $f(x) = (1 + x)^{\alpha}$ 的幂级数展开}

$1^\circ$ 当 $\alpha \leqslant -1$ 时, 有 $1 - \frac{\alpha+1}{k} \geqslant 1$ 对所有 $k \in \mathbb{N}$ 成立, 从而 $|C_\alpha^n| \geqslant 1$ 对所有 $n \in \mathbb{N}$ 成立, 所以 $f(x) = (1 + x)^{\alpha}$ 的幂级数展开在 $x = -1$ 处发散.

\pause
\vspace{0.5em}

$2^\circ$ 当 $-1 < \alpha < 0$ 时, 有 $0 < 1 - \frac{\alpha+1}{k} < 1$ 对所有 $k \in \mathbb{N}$ 成立.
\[
C_\alpha^n = (-1)^n \left( 1 - \frac{\alpha+1}{1} \right) \cdot \left( 1 - \frac{\alpha+1}{2} \right) \cdots \left( 1 - \frac{\alpha+1}{n} \right) =: (-1)^n \cdot u_n.
\]
由于 $\sum_{k=1}^\infty \left( - \frac{\alpha+1}{k} \right) = -\infty,$ 从而无穷乘积 $\prod_{k=1}^\infty \left( 1 - \frac{\alpha+1}{k} \right) = \prod_{k=1}^\infty (1 + a_k)$ 发散到 $0$, 从而 $u_n \searrow 0$ 当 $n \to \infty$. 所以
\begin{itemize}
\item 当 $x = 1$ 时, $\sum_{n=0}^\infty C_\alpha^n x^n = \sum_{n=0}^\infty (-1)^n u_n$ 为 Leibniz 级数, 收敛;
\pause
\item 当 $x = -1$ 时, $\sum_{n=0}^\infty C_\alpha^n x^n = \sum_{n=0}^\infty u_n$ 为正项级数. 并且
\[
\lim_{n \to \infty} \frac{u_{n+1}}{u_n} = \lim_{n \to \infty} \left( 1 - \frac{\alpha+1}{n+1} \right) = 1, ~~ \text{Cauchy, d'Alembert 判别法失效}.
\]
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{幂函数 $f(x) = (1 + x)^{\alpha}$ 的幂级数展开}

尝试 Raabe 判别法:
\[
\lim_{n \to \infty} n \left( \frac{u_n}{u_{n+1}} - 1 \right) = \lim_{n \to \infty} n \left( \frac{1}{1 - \frac{\alpha+1}{n+1}} - 1 \right) = \lim_{n \to \infty} n \cdot \frac{\alpha+1}{n+1 - (\alpha+1)} = \alpha + 1 \in (0, 1),
\]
所以由 Raabe 判别法可知 $\sum_{n=0}^\infty u_n$ 发散. 从而 $f(x) = (1 + x)^{\alpha}$ 的幂级数展开在 $x = -1$ 处发散.
\pause
综合可知, $-1 < \alpha < 0$ 时, $f(x) = (1 + x)^{\alpha}$ 的幂级数展开收敛域为 $[-1, 1)$.

\pause
\vspace{0.5em}

$3^\circ$ 当 $\alpha \geqslant 0$ 时, 关于 $u_n = |C_\alpha^n|$ 类似地有
\[
\frac{u_{n+1}}{u_n} = \left( 1 - \frac{\alpha+1}{n+1} \right) \xrightarrow{n \to \infty} 1, ~~ n \left( \frac{u_n}{u_{n+1}} - 1 \right) = (\alpha + 1) \cdot \frac{n}{n-\alpha} \xrightarrow{n \to \infty} \alpha + 1 > 1,
\]
故当 $x = \pm 1$ 时, 总有 $|C_\alpha^n x^n| = u_n,$ 级数绝对收敛.

\end{frame}

\frametitle{幂函数 $f(x) = (1 + x)^{\alpha}$ 的幂级数展开}

综合可知, 幂函数 $f(x) = (1 + x)^{\alpha}$ 在 $D$ 上有幂级数展开 $\sum_{n=0}^\infty C_\alpha^n x^n,$ 其中
\[
D = \begin{cases}
\mathbb{R}, & \alpha \in \mathbb{N}, \\
(-1, 1), & \alpha \leqslant -1, \\
[-1, 1), & -1 < \alpha < 0, \\
[-1, 1], & \alpha \geqslant 0 \text{ 且 } \alpha \notin \mathbb{N}.
\end{cases}
\]

\end{frame}

\begin{frame}
\frametitle{反三角函数的幂级数展开 | $\arcsin x$}

利用幂函数的幂级数展开相关结论
\[
(\arcsin x)' = (1 - x^2)^{-\frac{1}{2}} = \sum_{n=0}^\infty C_{-\frac{1}{2}}^n (-x^2)^n, ~~ \text{收敛域} ~ D = [-1, 1] (\text{\alert{注意}是 $x^2$}).
\]
结合逐项积分定理, 可以得到 $\arcsin x$ 的幂级数展开为
\begin{align*}
\arcsin x & = \int_0^x (1 - t^2)^{-\frac{1}{2}} \mathrm{d}t = \int_0^x \sum_{n=0}^\infty C_{-\frac{1}{2}}^n (-t^2)^n \mathrm{d}t = \sum_{n=0}^\infty C_{-\frac{1}{2}}^n \cdot \frac{(-1)^n}{2n+1} x^{2n+1} \\
& = \sum_{n=0}^\infty \prod_{k=1}^n \left( 1 - \frac{1}{2k} \right) \cdot \frac{1}{2n+1} x^{2n+1} = \sum_{n=0}^\infty \frac{(2n-1)!!}{(2n)!!} \cdot \frac{1}{2n+1} x^{2n+1}.
\end{align*}

\pause
\vspace{0.5em}

由于 $(\arcsin x)'$ 的幂级数展开的收敛域为 $[-1, 1]$, 逐项积分后收敛域不会缩小, 从而 $\arcsin x$ 的幂级数展开的收敛域也为 $[-1, 1]$.

当然, 也可以用数项级数的判别法验证 $x = \pm 1$ 处的情况, 从而验证 $\arcsin x$ 的幂级数展开在 $x = \pm 1$ 处收敛.
\qed

\end{frame}

\begin{frame}
\frametitle{反三角函数的幂级数展开 | $\arctan x$}

$(\arctan x)' = (1 + x^2)^{-1} = \sum_{n=0}^\infty C_{-1}^n x^{2n},$ 收敛域 $D = (-1, 1)$. 逐项积分可得
\begin{align*}
\arctan x & = \int_0^x (1 + t^2)^{-1} \mathrm{d}t = \int_0^x \sum_{n=0}^\infty C_{-1}^n t^{2n} \mathrm{d}t \\
& = \sum_{n=0}^\infty C_{-1}^n \cdot \frac{1}{2n+1} x^{2n+1} = \sum_{n=0}^\infty (-1)^n \prod_{k=1}^n \left( 1 - \frac{-1+1}{k} \right) \cdot \frac{1}{2n+1} x^{2n+1} \\
& = \sum_{n=0}^\infty \cdot \frac{(-1)^n}{2n+1} x^{2n+1}.
\end{align*}
在 $x = \pm 1$ 处, $\sum_{n=0}^\infty \cdot \frac{(-1)^n}{2n+1} x^{2n+1} = \sum_{n=0}^\infty \cdot \frac{(-1)^n}{2n+1} \cdot (-1)$ 为 Leibniz 级数, 收敛. 从而 $\arctan x$ 的收敛域为 $[-1, 1]$.
\pause
相较于 $(\arctan x)'$ 的幂级数展开的收敛域 $(-1, 1)$, 逐项积分后 $\arctan x$ 的幂级数展开的收敛域扩大到了 $[-1, 1]$.
\qed

\end{frame}

\begin{frame}
\frametitle{对数函数的幂级数展开}

对于对数函数 $f(x) = \ln (1 + x),$ 容易求得 $f^{(n)}(x) = (-1)^{n-1} (n-1)! (1 + x)^{-n},$ 从而有
\[
f(x) ~ \sim ~ \sum_{n=1}^\infty (-1)^{n-1} \frac{(n-1)!}{n!} \cdot (1)^{-n} \cdot x^n = \sum_{n=1}^\infty \frac{(-1)^{n-1}}{n} x^n.
\]
幂级数收敛域为 $D = (-1, 1]$. 余项 (Cauchy 形式) 有
\begin{align*}
r_n(x) & = \frac{f^{(n+1)}(c)}{n!} (x - c)^n x = \frac{(-1)^n n!}{n!} (1 + c)^{-n-1} (x - c)^n x \\
& = (-1)^n (1 + \theta x)^{-n-1} \cdot (1 - \theta)^n x^{n+1} = \frac{(-1)^n x^{n+1}}{1 + \theta x} \left( \underbrace{\frac{1 - \theta}{1 + \theta x}}_{\in (0, 1)} \right)^n \xrightarrow{n \to \infty} 0.
\end{align*}
所以 $\ln (1 + x)$ 有幂级数展开
\[
\ln (1 + x) = \sum_{n=1}^\infty \frac{(-1)^{n-1}}{n} x^n, ~~ \text{收敛域} ~ D = (-1, 1].
\]
\qed

\end{frame}

\begin{frame}
\frametitle{复合函数的幂级数展开}

\begin{eg}
求 $f(x) = \frac{1}{x^2}$ 在 $x = 1$ 处的幂级数展开 $\sum_{n=0}^\infty a_n (x - 1)^n.$
\end{eg}

\pause
\vspace{0.5em}

\startproof[bg=yellow, fg=red](解)
将 $f(x) = \frac{1}{x^2}$ 写成 $f(x) = \frac{1}{(1 + (x - 1))^2}$ 的形式, 不是我们熟悉的形式, 可能不太好求幂级数展开.
\pause
但是, 我们知道
\[
g(x) = \frac{1}{x} = \frac{1}{1 + (x - 1)} = \sum_{n=0}^\infty (-1)^n (x - 1)^n, ~~ |x - 1| < 1.
\]
\pause
在收敛域 $(0, 2)$ 内, 两边 (逐项) 求导可得
\[
f(x) = -g'(x) = -\sum_{n=1}^\infty (-1)^n n (x - 1)^{n-1} = \sum_{n=0}^\infty (-1)^n (n + 1) (x - 1)^n.
\]
收敛域相较 $g(x)$ 的收敛域不会扩大, 因此还是 $(0, 2)$.
\qed

\end{frame}

\begin{frame}
\frametitle{复合函数的幂级数展开}

\begin{eg}
求 $f(x) = \frac{1}{3 + 5x - 2x^2}$ 在 $x = 0$ 处的幂级数展开 $\sum_{n=0}^\infty a_n x^n.$
\end{eg}

\pause
\vspace{0.5em}

\startproof[bg=yellow, fg=red](解)
直接进行分解
\begin{align*}
f(x) & = \frac{1}{3 + 5x - 2x^2} = \frac{1}{(3 - x)(1 + 2x)} = \frac{1}{7} \cdot \frac{2(3 - x) + (1 + 2x)}{(3 - x)(1 + 2x)} \\
& = \frac{1}{7} \cdot \left( \frac{2}{1 + 2x} + \frac{1}{3 - x} \right) = \frac{1}{7} \cdot \left( 2 \sum_{n=0}^\infty (-2)^n x^n + \sum_{n=0}^\infty \frac{1}{3^{n+1}} x^n \right) \\
& = \frac{1}{7} \cdot \sum_{n=0}^\infty \left( (-1)^n \cdot 2^{n+1} + \frac{1}{3^{n+1}} \right) x^n.
\end{align*}
\pause
利用 Cauchy-Hadamard 定理, 可得幂级数展开的收敛半径为 $r = \frac{1}{2},$ 并可进一步验证其收敛域为 $\left(-\frac{1}{2}, \frac{1}{2}\right).$
\qed

\pause
\vspace{0.5em}

\begin{remark}
设幂级数 $\sum_{n=0}^\infty a_n x^n$ 与 $\sum_{n=0}^\infty b_n x^n$ 的收敛半径分别为 $r_a$ 与 $r_b$, 那么 $\sum_{n=0}^\infty (a_n + b_n) x^n$ 的收敛半径为 $r \geqslant \min \{ r_a, r_b \}.$
\end{remark}

\end{frame}

\begin{frame}
\frametitle{复合函数的幂级数展开}

\begin{question}
对一般的 $\frac{f(x)}{g(x)}$, 如何求其在 $x = 0$ 附近的幂级数展开?
\end{question}

\pause
\vspace{0.5em}

设 $f(x), g(x)$ 在 $x = 0$ 附近的幂级数展开分别为 $\sum_{n=0}^\infty a_n x^n$ 与 $\sum_{n=0}^\infty b_n x^n,$ 那么可以利用待定系数法, 设 $\frac{f(x)}{g(x)}$ 的幂级数展开为 $\sum_{n=0}^\infty c_n x^n,$ 那么
\[
\sum_{n=0}^\infty a_n x^n = \left(\sum_{n=0}^\infty b_n x^n \right) \cdot \left( \sum_{n=0}^\infty c_n x^n \right),
\]
比较两边的系数, 可以求得 $c_n$ 的递归关系, 从而求得 $c_n$ 的表达式.

\end{frame}

\begin{frame}
\frametitle{复合函数的幂级数展开}

\begin{eg}
求 $f(x) = \ln \frac{\sin x}{x}$ 在 $x = 0$ 处的幂级数展开 $\sum_{n=0}^\infty a_n x^n.$
\end{eg}

\only<2>{
\startproof[bg=yellow, fg=red](方法一)
利用 $\sin x$ 幂级数展开的结果, 有
\[
\frac{\sin x}{x} = \frac{1}{x} \cdot \sum_{n=0}^\infty (-1)^n \cdot \frac{x^{2n+1}}{(2n+1)!} = \sum_{n=0}^\infty (-1)^n \cdot \frac{x^{2n}}{(2n+1)!} = 1 + \sum_{n=1}^\infty (-1)^n \cdot \frac{x^{2n}}{(2n+1)!}.
\]
\pause
再利用 $\ln (1 + x)$ 的幂级数展开的结果, 有
\begin{align*}
f(x) & = \ln \left( 1 + \sum_{n=1}^\infty (-1)^n \cdot \frac{x^{2n}}{(2n+1)!} \right) \\
& = \sum_{n=1}^\infty \frac{(-1)^{n-1}}{n} \left( \sum_{k=1}^\infty (-1)^k \cdot \frac{x^{2k}}{(2k+1)!} \right)^n \\
& = \cdots
\end{align*}
}
\only<3>{
\startproof[bg=yellow, fg=red](方法二)
利用 $\frac{\sin x}{x}$ 的无穷乘积展开式 $\frac{\sin x}{x} = \prod_{n=1}^\infty \left( 1 - \frac{x^2}{n^2 \pi^2} \right),$ 可得
\begin{align*}
f(x) & = \ln \frac{\sin x}{x} = \ln \prod_{n=1}^\infty \left( 1 - \frac{x^2}{n^2 \pi^2} \right) = \sum_{n=1}^\infty \ln \left( 1 - \frac{x^2}{n^2 \pi^2} \right) \\
& = - {\color{red} \sum_{n=1}^\infty \sum_{k=1}^\infty} \frac{1}{k} \cdot \frac{x^{2k}}{n^{2k} \pi^{2k}} = - {\color{red} \sum_{k=1}^\infty \sum_{n=1}^\infty} \frac{1}{k} \cdot \frac{x^{2k}}{n^{2k} \pi^{2k}} \\
& = - \sum_{k=1}^\infty \frac{\zeta(2k)}{k \pi^{2k}} x^{2k}.
\end{align*}
}
\qed

\end{frame}

\begin{frame}
\frametitle{由 $\sin x$ 的无穷乘积展开式得到的计算公式}

之前对 $\sin x$ 的无穷乘积展开式 $\frac{\sin \pi x}{\pi x} = \prod_{n=1}^\infty \left( 1 - \frac{x^2}{n^2} \right)$ 取对数导数 $\frac{1}{\pi} \cdot \frac{\mathrm{d}}{\mathrm{d}x} \ln (\cdot)$ 得到了
\[
\cot \pi x = \frac{1}{\pi x} + \frac{1}{\pi} \sum_{n=1}^\infty \frac{2x}{x^2 - n^2} = \frac{1}{\pi x} + \frac{1}{\pi} \sum_{n=1}^\infty \left( \frac{1}{x - n} + \frac{1}{x + n} \right) = \frac{1}{\pi} \sum_{n \in \mathbb{Z}} \frac{1}{x + n}.
\]
\pause
另一方面,
\[
\cot \pi x = \frac{\cos \pi x}{\sin \pi x} = i \cdot \frac{e^{i \pi x} + e^{-i \pi x}}{e^{i \pi x} - e^{-i \pi x}} = i \cdot \frac{e^{2 i \pi x} + 1}{e^{2 i \pi x} - 1}.
\]
\pause
记 $t = t(x) = e^{2 i \pi x},$ 并令 $h_1(t) = \frac{1+t}{2(1-t)},$ 则有
\begin{empheq}[box={\fancymath[colback=cyan!30, drop lifted shadow, sharp corners]}]{equation*}
h_1(t) = \frac{1+t}{2(1-t)} = -\frac{1}{2 i \pi} \sum_{n \in \mathbb{Z}} \frac{1}{x + n} = -\frac{1}{2 i \pi} \left( \frac{1}{x} + \sum_{n=1}^\infty \frac{-2x}{n^2 - x^2} \right).
\end{empheq}

\end{frame}

\begin{frame}
\frametitle{$\frac{1}{n^2+1}$ 的求和}

取 $x = i,$ 那么 $t = e^{2 i \pi x} = e^{-2 \pi}.$ 代入 $h_1(t),$ 可得
\begin{align*}
\frac{1 + e^{-2 \pi}}{2(1 - e^{-2 \pi})} & = h_1(t) = -\frac{1}{2 i \pi} \left( \frac{1}{x} + \sum_{n=1}^\infty \frac{-2x}{n^2 - x^2} \right) = -\frac{1}{2 i \pi} \left( \frac{1}{i} + \sum_{n=1}^\infty \frac{-2i}{n^2 + 1} \right) \\
& = -\frac{1}{2 i \pi} \cdot \frac{1}{i} \left( 1 + 2 \sum_{n=1}^\infty \frac{1}{n^2 + 1} \right) = \frac{1}{2 \pi} \left( 1 + 2 \sum_{n=1}^\infty \frac{1}{n^2 + 1} \right).
\end{align*}
\pause
从而
\begin{empheq}[box={\fancymath[colback=cyan!30, drop lifted shadow, sharp corners]}]{equation*}
\sum_{n=1}^\infty \frac{1}{n^2 + 1} = \frac{\pi}{2} \cdot \frac{1 + e^{-2 \pi}}{1 - e^{-2 \pi}} - \frac{1}{2}.
\end{empheq}

\pause

\begin{frame}
\frametitle{Dirichlet $L$-函数特殊值与 $h_1(t)$}

对 $h_1(t) = - \frac{1}{2 i \pi} \sum_{n \in \mathbb{Z}} \frac{1}{x + n}$ 两边作用算子
\[
\left( t \cdot \frac{\mathrm{d}}{\mathrm{d}t} \right)^{r-1} = \left( e^{2 i \pi x} \cdot \frac{\mathrm{d}}{e^{2 i \pi x} \cdot 2 i \pi \mathrm{d}x} \right)^{r-1} = \left( \frac{\mathrm{d}}{2 i \pi \mathrm{d}x} \right)^{r-1} = \left(\frac{1}{2 i \pi}\right)^{r-1} \cdot \frac{\mathrm{d}^{r-1}}{\mathrm{d}x^{r-1}},
\]
可得
\begin{align*}
h_r(t) & := \left( t \cdot \frac{\mathrm{d}}{\mathrm{d}t} \right)^{r-1} h_1(t) = -\frac{1}{2 i \pi} \cdot \left(\frac{1}{2 i \pi}\right)^{r-1} \cdot \sum_{n \in \mathbb{Z}} \frac{(-1)^{r-1} (r-1)!}{(x + n)^r} \\
& = \left(\frac{1}{2 i \pi}\right)^{r} \cdot (r - 1)! \cdot \sum_{n \in \mathbb{Z}} \frac{1}{(x + n)^r}.
\end{align*}
\pause
现任取 $\chi$ 为 $\mod N$ 的 Dirichlet 特征, 则对 $r \in \mathbb{N}$ 有
\begin{align*}
L(r, \chi) & = \sum_{n=1}^\infty \frac{\chi(n)}{n^r} = \sum_{a \in \left( \mathbb{Z}/N\mathbb{Z} \right)^\times} \sum_{n \equiv a \mod N} \frac{\chi(n)}{n^r} = \sum_{a \in \left( \mathbb{Z}/N\mathbb{Z} \right)^\times} \chi(a) \sum_{k=0}^\infty \frac{1}{(kN + a)^r} \\
& = \sum_{a \in \left( \mathbb{Z}/N\mathbb{Z} \right)^\times} \chi(a) \sum_{k=0}^\infty \frac{1}{N^r \cdot \left( k + \frac{a}{N} \right)^r} = \frac{1}{2} \sum_{a \in \left( \mathbb{Z}/N\mathbb{Z} \right)^\times} \frac{\chi(a)}{(r-1)!} \cdot \left( -\frac{2 i \pi}{N} \right)^r \cdot h_r\left( e^{2 i \pi \frac{a}{N}} \right).
\end{align*}

\end{frame}

\begin{frame}
\frametitle{Dirichlet $L$-函数特殊值与 $h_1(t)$}

也就是说, $L(r, \chi)$ 可以通过 $h_r(t)$ 的值来计算:
\begin{empheq}[box={\fancymath[colback=cyan!30, drop lifted shadow, sharp corners]}]{equation*}
L(r, \chi) = \frac{1}{2 (r-1)!} \cdot \left( -\frac{2 i \pi}{N} \right)^r \cdot \sum_{a \in \left( \mathbb{Z}/N\mathbb{Z} \right)^\times} \chi(a) \cdot h_r\left( e^{2 i \pi \frac{a}{N}} \right).
\end{empheq}

\pause
\vspace{0.5em}

例如, 考虑 $\mod 4$ 的 Dirichlet 特征 $\chi$ 定义为 $\chi(1) = 1, \chi(3) = -1,$ 则 $L(s, \chi)$ 在 $s = 1$ 处的值为
\begin{align*}
L(r=1, \chi) & = 1 - \frac{1}{3} + \frac{1}{5} - \frac{1}{7} + \cdots \\
& = \left.\frac{1}{2 (r-1)!} \cdot \left( -\frac{2 i \pi}{N} \right)^r \cdot \sum_{a \in \left( \mathbb{Z}/N\mathbb{Z} \right)^\times} \chi(a) \cdot h_r\left( e^{2 i \pi \frac{a}{N}} \right) \right|_{r=1, N=4} \\
& = \frac{1}{2} \cdot \left( -\frac{i \pi}{2} \right) \cdot \left( h_1\left( e^{i \pi / 2} \right) - h_1\left( e^{3 i \pi / 2} \right) \right) = - \frac{i \pi}{4} \cdot \left( h_1(i) - h_1(-i) \right) \\
& = - \frac{i \pi}{4} \cdot \left( \frac{1 + i}{2(1 - i)} - \frac{1 - i}{2(1 + i)} \right) = \frac{\pi}{4}.
\end{align*}

\end{frame}
