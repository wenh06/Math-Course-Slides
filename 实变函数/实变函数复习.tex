\begin{frame}{集合的对等、势 | 可列集}

\begin{Def}
集合 $A$ 与 $B$ 被称为\alert{对等}的, 如果他们之间存在\alert{双射}, 记作 $A \sim B.$

\pause

也称 $A$ 与 $B$ \alert{等势}. 这种等价关系称为集合的\alert{势}, 记作 $|A|$ 或 $\operatorname{Card}(A).$
\end{Def}

\pause

\begin{remark}
势给出集合间的 (全) 序关系: 若 $A$ 与 $B$ 的某子集对等, 或等价地有单射 $A \hookrightarrow B,$ 则称 $A$ 的势小于等于 $B,$ 记作 $A \preccurlyeq B.$
\end{remark}

可列集: 与 $\mathbb{N}$ 等势的集合, 它们的势记作 $\aleph_0$ (即 $\aleph_0 = |\mathbb{N}|$). 例子:
$$\mathbb{N} \times \mathbb{N}, ~~ \mathbb{Q}, \dots$$

\pause

连续统: 与 $\mathbb{R}$ 等势的集合, 它们的势记作 $\mathfrak{c}.$ 有
$$\mathfrak{c} = |\mathcal{P}(\mathbb{N})| = 2^{\aleph_0}.$$


\end{frame}

\begin{frame}{开、闭集相关的概念与性质}

$\mathbb{R}$ 中开集 $G$: 所有点都是内点, 即
$$\forall x \in G, ~~ \exists ~ U(x, \delta) \subset G.$$

\pause

其他一些概念: 聚点 (极限点), 导集, 闭包, $\dots$

\pause

\begin{thm}[开集结构性定理]
有界开集 $G$ 总可以表示为至多可列多个不交开区间的并, 称为结构表示:
$$G = \bigcup_{i \in \Lambda} (a_i, b_i),$$
指标集 $\Lambda$ 至多可列, $(a_i, b_i)$ 称为构成区间.
\end{thm}

\pause

无界开集结论类似, 只要添加形如 $(-\infty, b), (a, +\infty)$ 的无限区间.

\pause
\vspace{0.8em}

比开、闭集更广义的集类:
\begin{center} 
$G_{\delta}$-集: 可列多个开集的交 \quad $F_{\sigma}$-集: 可列多个闭集的并

$G_{\delta\sigma}$-集, $F_{\sigma\delta}$-集, $\dots$
\end{center}

\end{frame}

\begin{frame}{Cantor 三分集的构造}

{\smaller[1]
\begin{align*}
F_1 & = F_{11} \cup F_{12} = \left[ 0, \dfrac{1}{3} \right] \cup \left[ \dfrac{2}{3}, 1 \right], \\
I_1 & = I_{11} = \left( \dfrac{1}{3}, \dfrac{2}{3} \right), \\
F_2 & = F_{21} \cup F_{22} \cup F_{23} \cup F_{24} = \left[ 0, \dfrac{1}{9} \right] \cup \left[ \dfrac{2}{9}, \dfrac{1}{3} \right] \cup \left[ \dfrac{2}{3}, \dfrac{7}{9} \right] \cup \left[ \dfrac{8}{9}, 1 \right], \\
I_2 & = I_{21} \cup I_{22} = \left( \dfrac{1}{9}, \dfrac{2}{9} \right) \cup \left( \dfrac{7}{9}, \dfrac{8}{9} \right), \\
& \vdots \\
F_n & = F_{n1} \cup F_{n2} \cup \cdots \cup F_{n2^{n}}, \\
I_n & = I_{n1} \cup I_{n2} \cup \cdots \cup I_{n2^{n-1}}, \\
& \vdots \\
G_0 & = \bigcup_{n=1}^{\infty} I_n, \\
P_0 & = \mathscr{C} G_0 = \bigcap_{n=1}^{\infty} F_n \longleftarrow \text{(康托三分集)}.
\end{align*}
}

\end{frame}

%------------------------------------------------

\begin{frame}{Cantor 三分集的性质}

\begin{itemize}
\item \alert{非空有界闭集}: $P_0 = \bigcap_{n=1}^{\infty} F_n,$ 作为闭集的交仍是闭集, 且包含所有区间端点
\item \alert{完全集}: 每个点都是聚点 (无孤立点), 即 $P_0' = P_0.$
\item \alert{测度为零}: 构造中移除的区间总长度为 
$$
m G_0 = \sum_{n=1}^{\infty} \sum_{k=1}^{2^{n-1}} m I_{nk}= \sum_{n=1}^{\infty} \frac{2^{n-1}}{3^n} = 1，
$$
故 $m P_0 = 1 - m G_0 = 0.$
\item \alert{不可数集}: $P_0$ 具有连续统的势 $\mathfrak{c}$ (与 $\mathbb{R}$ 等势).
\item \alert{无处稠密}: 在 $[0,1]$ 中不包含任何开区间.
\end{itemize}

\end{frame}

\begin{frame}{开、闭集测度}

对于 (有界) 开集 $\displaystyle G = \bigcup_{i \in \Lambda} (a_i, b_i),$ 可以定义\alert{开集测度}:
$$m G = \sum_{i \in \Lambda} b_i - a_i.$$

\pause

开集测度满足:
\begin{itemize}
\item \alert{单调性}: 若开集 $G_1 \subset G_2,$ 则 $m G_1 \leqslant m G_2.$
\item \alert{次可加性}: 对开集 $G_1, G_2,$ 有 $m\left( G_1 \cup G_2 \right) \leqslant m G_1 + m G_2.$
\item \alert{可列可加性}: 对任意可数个互不相交的开集 $G_1, G_2, \dots$ (即 $G_i \cap G_j = \emptyset$ 对 $i \neq j$), 有 $m\left( \bigcup_{i=1}^{\infty} G_i \right) = \sum_{i=1}^{\infty} m G_i.$
\end{itemize}

\pause
\vspace{1em}

开集测度可自然诱导出有界\alert{闭集测度}:
$$m F = b -a - m G,$$
其中闭集 $F \subset (a, b),$ $G = (a, b) \setminus F.$

\end{frame}

%------------------------------------------------

\begin{frame}{内、外测度}

\begin{itemize}
\item 外测度: $\displaystyle m^* E = \inf_{G \supset E} m G$
\item 内测度: $\displaystyle m_* E = \sup_{F \subset E} m F$
\end{itemize}

\pause

\begin{Def}[可测集]
集 $E \subset \mathbb{R}$ 称为可测集, 若满足
$$m^* E = m_* E \pause \alert{=: m E},$$
$m E$ 称为 $E$ 的测度.
\end{Def}

\end{frame}

\begin{frame}{各种意义下收敛的关系 | {\smaller[1] 一些典型的反例}}

\begin{itemize}[<+->]
\item 几乎处处收敛但不依测度收敛的例子 (注意, 必须有 $m E = \infty$):
$$f_n(x) = \chi_{[n, n + 1]}(x), \quad n \in \mathbb{N}.$$

\item 依测度收敛但不几乎处处收敛的例子: $E = [0, 1],$
$$f_n(x) = \chi_{\left[ \frac{i}{2^k}, \frac{i + 1}{2^k} \right]}(x), \quad n = 2^k + i, \quad 0 \leqslant i < 2^k.$$
以上也是强收敛但不几乎处处收敛的例子.

\item 依测度收敛但不强收敛的例子 (稍微修改上例):
$$f_n(x) = {\color{red} 2^k} \chi_{\left[ \frac{i}{2^k}, \frac{i + 1}{2^k} \right]}(x), \quad n = 2^k + i, \quad 0 \leqslant i < 2^k.$$
依测度收敛 (到 $0$) 但不弱收敛的例子 (取 $g(x) = 1$).
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}{各种意义下收敛的关系 | {\smaller[1] 一些典型的反例}}

\begin{itemize}[<+->]
\item 一致收敛但不弱收敛 (也不强收敛) 的例子 (必须有 $m E = \infty$):
$$f_n(x) = \dfrac{1}{n} \chi_{[1, e^n]}(x).$$
容易验证 $f_n(x)$ 一致收敛到 $0.$ 设 $p, q > 1$ 为共轭数, 则 $f_n \in L^p(\mathbb{R}).$ 取 $\displaystyle g(x) = \dfrac{\chi_{[1, +\infty)}(x)}{x},$ 则 $g \in L^q(\mathbb{R}),$ 但是 $\displaystyle \int_{\mathbb{R}} f_n(x) g(x) ~ \mathrm{d} x = \dfrac{1}{n} \int_{1}^{e^n} \dfrac{1}{x} ~ \mathrm{d} x = 1.$

\item 弱收敛但不一致收敛的例子: $E = [0, 1),$
$$f_n(x) = x^n, \quad n \in \mathbb{N}.$$

\item 弱收敛但不强收敛的例子: $E = (0, \pi),$
$$f_n(x) = \sin(nx), \quad n \in \mathbb{N}.$$
\end{itemize}

\end{frame}
