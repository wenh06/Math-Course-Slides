\section{映射}

%------------------------------------------------

\begin{frame}{映射}

\begin{Def}[映射]
设 $X, Y$ 是两个集合. 若对每个 $x \in X$, 按照某种规则 $f$, 都有唯一确定的 $y \in Y$ 与之对应, 则称 $f$ 为从 $X$ 到 $Y$ 的一个\alert{映射} (mapping), 记为
\[
f: X \to Y, \quad x \mapsto f(x) = y.
\]
\end{Def}

\pause

\begin{block}{基本术语}
\begin{itemize}
  \item \alert{定义域} (domain): $\operatorname{Dom}(f) = X$
  \item \alert{值域} / \alert{像} (range / image): $f(X) = \{ f(x) : x \in X \}$
  \item $f(x)$ 称为 $x$ 在 $f$ 下的\alert{像}, $x$ 称为 $f(x)$ 在 $f$ 下的\alert{原像} (preimage)
\end{itemize}
\end{block}

\pause

\begin{remark}
两个法则 $f, g: X \to Y$ 给出方式不同, 但可能有同一效果, 即 $f(x) = g(x)$ 对所有 $x \in X$ 都成立, 则认为 $f, g$ 作为映射是\alert{相等}的.

\pause

例如 $X = Y = \mathbb{F}_2 = \{\bar{0}, \bar{1}\}$ 上的映射 $f(x) = x$ 与 $g(x) = x^2,$ 容易验证 $f(\bar0) = g(\bar{0}) = \bar{0},$ $f(\bar{1}) = g(\bar{1}) = \bar{1},$ 故 $f$ 与 $g$ 是同一个映射, 但他们作为 $\mathbb{F}_2[x]$ 中的元素 (多项式) 却不相等.
\end{remark}

\end{frame}

%------------------------------------------------

\begin{frame}{像与原像}

设 $f: X \to Y$ 为映射, $A \subset X,$ $B \subset Y:$

\begin{Def}
\begin{itemize}
  \item $A$ 的\alert{像}: $f(A) = \{ f(x) : x \in A \} \subset Y$
  \item $B$ 的\alert{原像}: $f^{-1}(B) = \{ x \in X : f(x) \in B \} \subset X$
\end{itemize}
\end{Def}

\pause
\vspace{0.3em}

\begin{prop}[原像与集合运算]
设 $\{ B_\alpha \}_{\alpha \in I} \subset Y,$ $B \subset \widetilde{B} \subset Y,$ 则
\begin{align*}
f^{-1}\!\left( \bigcup_{\alpha \in I} B_\alpha \right) = \bigcup_{\alpha \in I} f^{-1}(B_\alpha), & \hspace{1.5em}
f^{-1}\!\left( \bigcap_{\alpha \in I} B_\alpha \right) = \bigcap_{\alpha \in I} f^{-1}(B_\alpha), \\
f^{-1}(\mathscr{C} B) = \mathscr{C}\, f^{-1}(B), & \hspace{1.5em}
f^{-1}(B) \subset f^{-1}(\widetilde{B}).
\end{align*}
\end{prop}

\pause

\begin{remark}
原像运算与集合的\alert{并、交、补均相容}.
\end{remark}

\end{frame}

%------------------------------------------------

\begin{frame}{像与原像}

\begin{prop}[像与集合运算]
设 $\{ A_\alpha \}_{\alpha \in I} \subset X,$ 则
\[
f\!\left( \bigcup_{\alpha \in I} A_\alpha \right) = \bigcup_{\alpha \in I} f(A_\alpha), \quad
f\!\left( \bigcap_{\alpha \in I} A_\alpha \right) \subset \bigcap_{\alpha \in I} f(A_\alpha).
\]
\end{prop}

\pause
\vspace{0.5em}

\begin{remark}
像运算与集合的\alert{并相容}, 但\alert{交不相容}.

\pause

{\color{red}\ding{43}} 反例: $f: \mathbb{R} \to \mathbb{R}, x \mapsto x^2,$ $A_1 = (-\infty, 0), A_2 = (0, \infty).$ 则 $f(A_1 \cap A_2) = f(\emptyset) = \emptyset,$ 但 $f(A_1) \cap f(A_2) = [0, \infty)$.

\pause

但如果 $f$ 是\alert{单射}, 则像运算与集合的交也相容, 即 $f\!\left( \bigcap_{\alpha \in I} A_\alpha \right) = \bigcap_{\alpha \in I} f(A_\alpha)$.
\end{remark}

\end{frame}

%------------------------------------------------

\begin{frame}{满射、单射与双射；逆映射; 映射的复合}

设 $f: X \to Y$ 为映射

\vspace{0.4em}

\begin{itemize}
\item \alert{单射} (injection / one-to-one): $\forall~x_1, x_2 \in X,\ f(x_1) = f(x_2) \Rightarrow x_1 = x_2$
\item \alert{满射} (surjection / onto): $f(X) = Y$, 即 $\forall~y \in Y,\ \exists~x \in X$ 使 $f(x) = y$
\item \alert{双射} (bijection): 既是单射又是满射, 也称\alert{一一对应} (one-to-one correspondence)
\end{itemize}

\pause
\vspace{0.5em}

\begin{block}{逆映射}
若 $f: X \to Y$ 是双射, 则对每个 $y \in Y$ 存在唯一 $x \in X$ 使 $f(x) = y$. 由此定义 $f$ 的\alert{逆映射}
\[
f^{-1}: Y \to X, \quad y \mapsto x \iff f(x) = y.
\]
此时 $f^{-1} \circ f = \mathrm{id}_X,$ $f \circ f^{-1} = \mathrm{id}_Y$. $f^{-1}$ 也是双射, 且 $(f^{-1})^{-1} = f$.
\end{block}

\pause
\vspace{0.5em}

\begin{block}{映射的复合}
设 $f: X \to Y,$ $g: Y \to Z$ 为映射, 由关系 $x \mapsto g(f(x))$ 给出的映射称为 $g$ 与 $f$ 的\alert{复合} (composition), 记作
\[
g \circ f: X \to Z, \quad x \mapsto g(f(x)).
\]
\end{block}

\end{frame}

%------------------------------------------------

\begin{frame}{限制、扩张与特征函数}

\begin{block}{限制与扩张}
设 $f: X \to Y$ 为映射, $A \subset X$:
\begin{itemize}
\item $f$ 在 $A$ 上的\alert{限制} (restriction): $f|_A: A \to Y,$ 定义为 $f|_A(x) = f(x)$
\item 若 $g: Z \to Y$ 满足 $Z \supset X$ 且 $g|_X = f,$ 则称 $g$ 是 $f$ 的一个\alert{扩张} (extension)
\end{itemize}
\end{block}

\pause
\vspace{0.5em}

\begin{block}{特征函数 (示性函数)}
设 $A \subset X,$ $A$ 的\alert{特征函数} (characteristic function) 为
\[
\chi_A(x) = \begin{cases} 1, & x \in A, \\ 0, & x \notin A. \end{cases}
\]
\end{block}

\pause
\vspace{0.3em}

特征函数的基本运算性质:
\[
\chi_{A \cap B} = \chi_A \cdot \chi_B, \quad
\chi_{A \cup B} = \chi_A + \chi_B - \chi_A \chi_B, \quad
\chi_{\mathscr{C} A} = 1 - \chi_A, \quad
\chi_{A \setminus B} = \chi_A (1 - \chi_B).
\]

\end{frame}

\section{对等与可列集}

%------------------------------------------------

\begin{frame}{集合的对等}

\begin{Def}[对等]
若集合 $A$ 与 $B$ 之间存在\alert{双射}, 则称 $A$ 与 $B$ \alert{对等} (equipotent) 或\alert{等势}, 记作 $A \sim B$.
\end{Def}

\pause
\vspace{0.3em}

对等是集合之间的\alert{等价关系}:
\begin{itemize}
  \item \alert{自反性}: $A \sim A$ (恒等映射 $\mathrm{id}_A$)
  \item \alert{对称性}: $A \sim B \Rightarrow B \sim A$ (逆映射 $f^{-1}$)
  \item \alert{传递性}: $A \sim B,\ B \sim C \Rightarrow A \sim C$ (复合 $g \circ f$)
\end{itemize}

\pause
\vspace{0.3em}

\begin{Def}[势与序]
$A$ 所在的等价类称为 $A$ 的\alert{势} (cardinality), 记作 $|A|$ 或 $\operatorname{Card}(A)$.

若存在单射 $A \hookrightarrow B$, 称 $A$ 的势\alert{不超过} $B$, 记 $|A| \leqslant |B|$ (或 $A \preccurlyeq B$).
\end{Def}

\pause
\vspace{0.3em}

\begin{eg}
$\mathbb{R}$ 与开区间 $(-1, 1)$ 通过双射 $f: \mathbb{R} \to (-1, 1), x \mapsto \tanh(x) = \frac{e^x - e^{-x}}{e^x + e^{-x}}$ 对等.
\pause
当然, 这样的映射并不唯一, 例如还可以取 $\frac{x}{1 + |x|}.$
\end{eg}

\end{frame}

%------------------------------------------------

\begin{frame}{可列集}

\begin{Def}[可列集]
与自然数集 $\mathbb{N}$ 对等的集合称为\alert{可列集} (countable set), 其势记为 $\aleph_0 = |\mathbb{N}|$.

有限集与可列集统称\alert{至多可列集}.
\end{Def}

\pause
\vspace{0.3em}

\begin{eg}
以下均为可列集:
\begin{itemize}
\item $\mathbb{Z}$: 按 $0, 1, -1, 2, -2, 3, -3, \ldots$ 排列, 建立与 $\mathbb{N}$ 的双射
\item $\mathbb{N} \times \mathbb{N}$: \alert{对角线法}, $(m, n) \mapsto \dfrac{(m+n-2)(m+n-1)}{2} + m$
\item $\mathbb{Q}$: 因 $\mathbb{Q} \subset \mathbb{Z} \times \mathbb{N}^+$ (以分子/分母编码), 再利用 $\mathbb{N} \times \mathbb{N} \sim \mathbb{N}$
\end{itemize}
\end{eg}

\pause
\vspace{0.3em}

\begin{remark}
可列集可以对元素进行编号, 形成无穷序列 $\{a_1, a_2, a_3, \ldots\}$. 可列集的子集是至多可列的, 可以说``最小''的无限集是可列集.
\end{remark}

\end{frame}

\section{无限集的性质}

%------------------------------------------------

\begin{frame}{无限集的性质 (一)}

\begin{thm}
\begin{enumerate}
\item 无限集必有\alert{可列子集}.
\item 可列个可列集之\alert{并}仍是可列集.
\end{enumerate}
\end{thm}

\pause
\vspace{0.3em}

\startproof[bg=yellow, fg=red](证明要点)
\vspace{0.3em}

\textbf{(1)} 设 $A$ 为无限集. 取 $a_1 \in A$; 归纳地, 已取 $a_1, \ldots, a_n$, 因 $A$ 无限故 $A \setminus \{a_1, \ldots, a_n\} \neq \emptyset$, 再取 $a_{n+1}$. 如此得到可列子集 $\{a_n\}_{n \geqslant 1} \subset A$.

\pause
\vspace{0.5em}

\textbf{(2)} 设 $A_1, A_2, \ldots$ 均可列, $A_k = \{a_{k1}, a_{k2}, \ldots\}$. 将其排成无穷矩阵
\[
\begin{matrix}
a_{11} & a_{12} & a_{13} & \cdots \\
a_{21} & a_{22} & a_{23} & \cdots \\
a_{31} & a_{32} & a_{33} & \cdots \\
\vdots & \vdots & \vdots & \ddots
\end{matrix}
\]
按\alert{对角线顺序}枚举: \quad $a_{11}, ~ | ~ a_{12},\ a_{21}, ~ | ~ a_{13},\ a_{22},\ a_{31}, ~ | ~\ldots$
\newline
故 $\bigcup_{k=1}^\infty A_k$ 至多可列; 若某个 $A_k$ 是无限可列集则并集恰为可列集.

\end{frame}

%------------------------------------------------

\begin{frame}{无限集的性质 (二)}

\begin{thm}
无限集 $A$ 与其某个\alert{真子集}对等 (这也是无限集的等价刻画).
\end{thm}

\pause
\vspace{0.3em}

\startproof[bg=yellow, fg=red](证明)
\vspace{0.3em}

由性质 (一) 中的结论 1, $A$ 有可列子集 $S = \{a_1, a_2, a_3, \ldots\}$. 取真子集 $B = A \setminus \{a_1\}$, 定义双射
\[
f: A \to B, \quad f(x) = \begin{cases} a_{n+1}, & x = a_n \in S, \\ x, & x \in A \setminus S. \end{cases}
\]
$f$ 将 $S$ 中元素依次平移 ($a_1 \mapsto a_2 \mapsto a_3 \mapsto \cdots$), $A \setminus S$ 部分恒等, 故 $f: A \xrightarrow{\sim} A \setminus \{a_1\} = B$.
\qed

\pause
\vspace{0.3em}

\begin{remark}
有限集\alert{不具备}此性质: 有限集不可能与自身的真子集对等.
\end{remark}

\end{frame}

\section{Cantor-Bernstein 定理}

%------------------------------------------------

\begin{frame}{集在映射下的分解引理}

\begin{lem}[集在映射下的分解引理]
设有映射 $f: X \to Y,$ $g: Y \to X,$ 则有分解
\[
X = A \cup \widetilde{A}, \quad Y = B \cup \widetilde{B},
\]
使得 $f(A) = B,$ $g(\widetilde{B}) = \widetilde{A},$ $A \cap \widetilde{A} = B \cap \widetilde{B} = \emptyset.$
\end{lem}

\pause
\vspace{0.3em}

\startproof[bg=yellow, fg=red](证明)
称 $C \subset X$ 为\alert{分离集}, 若 $g(Y \setminus f(C)) \cap C = \emptyset$. 设 $\mathscr{F}$ 为所有分离集构成的集族, 令
\[
A = \bigcup_{C \in \mathscr{F}} C.
\]

\pause

{\color{red}\ding{43}} \textbf{$A$ 是分离集}: 不妨设 $Y \neq f(A),$ 任取 $y \in Y \setminus f(A)$, $x = g(y)$. 若 $x \in A$, 则 $\exists ~ C \in \mathscr{F}$ 使得 $x \in C$. 由 $y \notin f(A) \supset f(C)$ 知 $y \in Y \setminus f(C)$, 故 $g(y) \notin C$ (因 $C \in \mathscr{F}$), 即 $x \notin C$, 矛盾. 故 $g(Y \setminus f(A)) \cap A = \emptyset$, $A \in \mathscr{F}$.

\pause
\vspace{0.3em}

{\color{red}\ding{43}} \textbf{等号成立}: 取 $B = f(A)$, 由 $A \in \mathscr{F}$ 得 $g(\widetilde{B}) = g(Y \setminus f(A)) \subset \widetilde{A} = X \setminus A$. 任取 $x \in \widetilde{A}$. 若 $A \cup \{x\}$ 仍是分离集, 则可加入 $A$, 与极大性矛盾. 故 $\exists\, y \in Y \setminus f(A \cup \{x\}) \subset \widetilde{B}$ 使 $g(y) \in A \cup \{x\}$. 由 $A \in \mathscr{F}$ 得 $g(y) \notin A$, 故 $g(y) = x$, 即 $x \in g(\widetilde{B})$.
\qed

\end{frame}

%------------------------------------------------

\begin{frame}{Cantor-Bernstein 定理}

\begin{thm}[Cantor-Bernstein 定理]
设存在单射 $f: A \hookrightarrow B$ 和单射 $g: B \hookrightarrow A$ (即 $A \preccurlyeq B$ 且 $B \preccurlyeq A$), 则 $A \sim B$.
\end{thm}

\pause
\vspace{0.3em}

\startproof[bg=yellow, fg=red](由分解引理推导)
对两个单射 $f: A \hookrightarrow B,$ $g: B \hookrightarrow A$ 应用分解引理, 得
\[
A = A_1 \cup \widetilde{A}_1, \quad B = B_1 \cup \widetilde{B}_1, \quad A_1 \cap \widetilde{A}_1 = B_1 \cap \widetilde{B}_1 = \emptyset,
\]
满足 $f(A_1) = B_1,$ $g(\widetilde{B}_1) = \widetilde{A}_1.$

\pause

\begin{itemize}
\item $f|_{A_1}: A_1 \to B_1$ 单射（$f$ 单射的限制）且满射（由 $f(A_1)=B_1$）, 故 $f|_{A_1}: A_1 \xrightarrow{\;\sim\;} B_1$ 是双射.
\item $g|_{\widetilde{B}_1}: \widetilde{B}_1 \to \widetilde{A}_1$ 单射且满射（由 $g(\widetilde{B}_1)=\widetilde{A}_1$）, 故 $g|_{\widetilde{B}_1}: \widetilde{B}_1 \xrightarrow{\;\sim\;} \widetilde{A}_1$ 是双射, 进而 $g^{-1}|_{\widetilde{A}_1}: \widetilde{A}_1 \xrightarrow{\;\sim\;} \widetilde{B}_1$.
\end{itemize}

\pause

定义 $h: A \to B$ 为 $h(x) = \begin{cases} f(x), & x \in A_1, \\ g^{-1}(x), & x \in \widetilde{A}_1. \end{cases}$ 由 $B_1 \cap \widetilde{B}_1 = \emptyset$, $h$ 是 $A \to B$ 的双射.
\qed

\pause
\vspace{0.4em}

\begin{remark}
Cantor-Bernstein 定理保证了势的\alert{反对称性}: $|A| \leqslant |B|$ 且 $|B| \leqslant |A|$ $\Rightarrow$ $|A| = |B|$, 即势构成集合的\alert{全序}.
\end{remark}

\end{frame}

\section{不可列集}

%------------------------------------------------

\begin{frame}{$[0,1]$ 区间不可列}

\begin{thm}
区间 $[0,1]$ 不是可列集, 其势为 $\aleph = \mathfrak{c} = |\mathbb{R}|$.
\end{thm}

\pause

\startproof[bg=yellow, fg=red]
假设 $[0,1]$ 可列, 设 $[0,1] = \{ x_1, x_2, x_3, \ldots \}$. 将 $[0,1]$ 三等分为三个闭子区间:
\[
\left[0, \tfrac{1}{3}\right], \quad \left[\tfrac{1}{3}, \tfrac{2}{3}\right], \quad \left[\tfrac{2}{3}, 1\right].
\]
$x_1$ 至多属于其中两个, 故可取一个不含 $x_1$ 的子区间 $I_1$. 其长度 $|I_1| = \tfrac{1}{3}$.

\pause

再将 $I_1$ 三等分, 取子区间 $I_2 \subset I_1$ 使 $x_2 \notin I_2$. 如此递推, 得到闭区间列
\[
I_1 \supset I_2 \supset I_3 \supset \cdots, \quad x_n \notin I_n, \quad |I_n| = \left(\tfrac{1}{3}\right)^n \to 0.
\]

\pause

由\alert{闭区间套定理}: $\exists\, \xi \in \displaystyle\bigcap_{n=1}^{\infty} I_n \subset [0,1].$

对每个 $n$: $\xi \in I_n$ 但 $x_n \notin I_n$, 故 $\xi \neq x_n$. 这与 $[0,1] = \{ x_1, x_2, \ldots \}$ 矛盾.
\qed

\end{frame}

%------------------------------------------------

\begin{frame}{Cantor 定理：集合与其幂集不对等}

\begin{thm}[Cantor 定理]
对任意集合 $A,$ $A$ 与其幂集 $\mathscr{P}(A)$ 不对等, 即 $|A| < |\mathscr{P}(A)|$.
\end{thm}

\pause
\vspace{0.3em}

\startproof[bg=yellow, fg=red]
\vspace{0.3em}

{\color{red}\ding{43}} $|A| \leqslant |\mathscr{P}(A)|$: 单射 $x \mapsto \{x\}$ 存在.

\pause
\vspace{0.3em}

{\color{red}\ding{43}} 假设存在\alert{满射} $f: A \to \mathscr{P}(A)$, 令
\[
D = \{ x \in A : x \notin f(x) \} \in \mathscr{P}(A).
\]
因 $f$ 是满射, $\exists~d \in A$ 使 $f(d) = D$.

\pause

\begin{itemize}
\item 若 $d \in D$: 由 $D$ 的定义, $d \notin f(d) = D$, 矛盾.
\item 若 $d \notin D$: 由 $D$ 的定义, $d \in f(d) = D$, 矛盾.
\end{itemize}

故不存在满射 $A \to \mathscr{P}(A)$, 从而 $|A| < |\mathscr{P}(A)|$.
\qed

\pause
\vspace{0.3em}

\begin{remark}
由此得到势的无穷严格递增链: $\aleph_0 < |\mathscr{P}(\mathbb{N})| < |\mathscr{P}(\mathscr{P}(\mathbb{N}))| < \cdots,$ 其中 $|\mathscr{P}(\mathbb{N})| = \mathfrak{c}$.
\end{remark}

\end{frame}
