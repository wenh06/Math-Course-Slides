\begin{frame}

\begin{algorithm}[H]
\SetAlgoLined
\DontPrintSemicolon
\SetKwInOut{Input}{Input}
\SetKwInOut{Output}{Output}

\Input{训练集$\left\{(y_i,\mathbf{x}_i)\right\}_{i=1}^N$}
\Input{可微凸损失函数$L(y,y_{pred})$}
\Input{迭代次数$M$}
% \Input{weak learner $f_m(\mathbf{x}), m = 1,\ldots,M$}
% initialize model with a constant value
初始化模型$F_0(\mathbf{x}) = \gamma_0 = \argmin\limits_{\gamma} \sum\limits_{1 \leqslant i \leqslant N}L(y_i,\gamma)$\;
\For{$m = 1$ \KwTo $M$}
{
计算$g_{im} = -\partial_2 L(y_i, F_{m-1}(\mathbf{x}_i)), \quad i = 1, \ldots, N$\;
将$\left\{(g_{im},\mathbf{x}_i)\right\}_{i=1}^N$作为训练集训练 weak learner $f_m(\mathbf{x})$\;
计算$\gamma_m = \argmin\limits_{\gamma} \sum\limits_{1 \leqslant i \leqslant N} L(y_i, F_{m-1}(\mathbf{x}_i) + \gamma f_m(\mathbf{x}_i))$\;
令$F_m(\mathbf{x}) = F_{m-1}(\mathbf{x}) + \gamma_m f_m(\mathbf{x})$
}
\Output{最终模型$F(\mathbf{x}) = F_M(\mathbf{x})$}
\caption{一般性的梯度提升方法（Gradient Boosting Machine）}
\end{algorithm}

\end{frame}

\section{原理简述} 
%\subsection{Subsection Example} % A subsection can be created just before a set of slides with a common theme to further break down your presentation into chunks

%------------------------------------------------

\begin{frame}

对于损失函数的期望（即在训练集上的求和）$$\mathbb{E}_{y,\mathbf{x}}[L(y, F(\mathbf{x}))] = \sum\limits_{i=1}^N L(y_i, F(\mathbf{x}_i)),$$
在第$m$步，我们考虑的是，在$F_{m-1}$上加上一个什么样的$f_m$，能使损失函数的期望进一步降低（比第$m-1$步的时候），即
\begin{align*}
f_m & = \argmin_{f} \sum\limits_{i=1}^N L(y_i, F_{m-1}(\mathbf{x}_i) + f(\mathbf{x}_i)) \\
& \approx \argmin_{f} \sum\limits_{i=1}^N \left( L(y_i, F_{m-1}(\mathbf{x}_i)) + \partial_2 L(y_i, F_{m-1}(\mathbf{x}_i)) f(\mathbf{x}_i) \right)
\end{align*}

\end{frame}

%------------------------------------------------

\begin{frame}

于是根据梯度下降法，自然的选择是取$f_m$使得
$$f_m(\mathbf{x}_i) = -\gamma_m \partial_2 L(y_i, F_{m-1}(\mathbf{x}_i)).$$

也就是原算法迭代步骤第4行所说的，将
$$\left\{(\partial_2 L(y_i, F_{m-1}(\mathbf{x}_i)), \mathbf{x}_i) \right\}_{i=1}^N$$
作为训练集训练 weak learner $f_m(\mathbf{x})$，再去搜索前面的系数$\gamma_m$。

\end{frame}
