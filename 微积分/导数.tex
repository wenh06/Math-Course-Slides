\section{导数的定义} % Sections can be created in order to organize your presentation into discrete blocks, all sections and subsections are automatically printed in the table of contents as an overview of the talk
%------------------------------------------------
\begin{frame}

\begin{Def}
设函数$f$在点$x$的近旁有定义，若极限
$$\lim\limits_{h\to 0} \dfrac{f(x+h)-f(x)}{h}$$
存在且有限，则称这个数为函数$f$在点$x$处的{\bfseries 导数（derivative）}，记作$f'(x),$ 或$\dfrac{df(x)}{dx}$。此时，称函数$f$在点$x$处可微，
$$df = f'(x)dx$$
被称为函数$f$在点$x$处的微分（differential）。
\end{Def}

\pause

\begin{remark}
\begin{itemize}
\item 函数在一点的导数若存在则唯一。
\item 函数在一点可微，则在这点连续；反之不成立。
\end{itemize}
\end{remark}

\end{frame}

\section{求导法则}

\subsection{求导的四则运算}

%------------------------------------------------

\begin{frame}

设函数$f,g$都在点$x$处可微，那么
\begin{itemize}
\item<2-> $f+g$在$x$处可微，且
$$(f+g)'(x) = (f'+g')(x);$$
\item<3-> $f\cdot g$在$x$处可微，且
$$(f\cdot g)'(x) = f'(x)\cdot g(x) + f(x)\cdot g'(x);$$
\item<4-> 若$g(x)\neq0$，则$\dfrac{f}{g}$在$x$处可微，且
$$\dfrac{f'(x)\cdot g(x) - f(x)\cdot g'(x)}{g^2(x)}$$
\end{itemize}

\end{frame}

%------------------------------------------------

\subsection{复合函数求导：链式法则}

%------------------------------------------------

\begin{frame}

\begin{thm}[链式法则，Chain Rule]
设函数$y=f(x)$在点$x_0$处可微，函数$z = g(y)$在点$y_0 = f(x_0)$处可微，那么复合函数$g\circ f$在点$x_0$处可微，并且有
$$(g\circ f)'(x_0) = g'(f(x_0))\cdot f'(x_0).$$
\end{thm}

\pause

使用Leibniz的记号，以上定理可以写作
$$\dfrac{dz}{dx} = \dfrac{dz}{dy} \cdot \dfrac{dy}{dx}$$

\pause

\begin{cor}
对于$n$个可微函数$y_1 = f_1(x), \ldots, y_n = f_n(y_{n-1})$的复合函数$(f_n\circ\cdots\circ f_1)(x),$ 有
$$(f_n \circ \cdots \circ f_1)'(x) = f_n'(y_{n-1}) \cdots f_1'(x).$$
\end{cor}

\end{frame}

%------------------------------------------------

\begin{frame}

\begin{eg}
\centering
求函数$y = \sin\left(\ln\dfrac{x}{1+x^2}\right)$的导数，$x > 0.$
\end{eg}

\pause

令
$$y_1 = \dfrac{x}{1+x^2},\quad  y_2 = \ln y_1,\quad  y_3 = \sin y_2,$$
它们分别有导数
$$y_1' = \dfrac{1-x^2}{(1+x^2)^2},\quad  y_2' = \dfrac{1}{y_1},\quad y_3' = \cos y_2.$$
有$y(x) = y_3 \circ y_2 \circ y_1(x),$ 于是根据复合函数求导的链式法则
\begin{align*}
y' & = y_3'(y_2)\cdot y_2'(y_1)\cdot y_1'(x) \\
& = \left(\cos\left(\ln\dfrac{x}{1+x^2}\right)\right) \cdot \left(\dfrac{x}{1+x^2}\right)^{-1} \cdot \dfrac{1-x^2}{(1+x^2)^2}
\end{align*}

\end{frame}

\subsection{反函数求导}

%------------------------------------------------

\begin{frame}

\begin{thm}<1->
设函数$f$在点$x_0$的某个邻域内有反函数$f^{-1}$，且它们分别在点$x_0$与$y_0=f(x_0)$处连续。若$f$在点$x_0$处可微且$f'(x_0) \neq 0,$ 那么$f^{-1}$也在点$y_0$处可微且
$$(f^{-1})'(y_0) = \dfrac{1}{f'(x_0)}.$$
\end{thm}

\begin{eg}<2->
\centering
求反三角函数$y = \arcsin x$的导数，$x\in(-\dfrac{\pi}{2}, \dfrac{\pi}{2}).$
\end{eg}

\pause

$y = \arcsin x$与$x = \sin y$互为反函数，于是根据反函数求导法则有
$$y' =  \arcsin' x = \dfrac{1}{\sin' y} = \dfrac{1}{\cos y} = \dfrac{1}{\sqrt{1-x^2}}.$$


\end{frame}

%------------------------------------------------

\subsection{隐函数求导初步}

%------------------------------------------------

\begin{frame}

\begin{block}{隐函数}<1->
考虑$f(x,y) = 0.$ 在某个区间$I$上，对任意$x_0\in I,$ 存在唯一的$y_0$使得$f(x_0,y_0) = 0$成立，那么称$f(x,y) = 0$在$I$上确定了一个{\bfseries 隐函数}$y = y(x).$
\end{block}

\begin{block}{隐函数求导法}<2->
把$f(x,y(x)) = 0$视作关于$x$的等式，两边同时关于$x$求导仍是等式。关于$x$求导用上之前讲过的导数四则运算，链式法则即可。
\end{block}

\begin{eg}<3->
考虑$e^x + \ln x = y + \varepsilon \sin y, 0 < |\varepsilon| < 1.$ 两边同时对$x$求导有$e^x + \frac{1}{x} = y' + \varepsilon y'\cos y,$ 于是
$$y' = \dfrac{e^x + \frac{1}{x}}{1+\varepsilon\cos y}$$
\end{eg}

\end{frame}

\subsection{高阶导数}

%------------------------------------------------

\begin{frame}

如果函数$f$在某个区间$I$上每点都可微，那么有新的函数$f'$，他在每点$x\in I$的取值等于$f$在$x$处的导数。称他为{\bfseries 导函数}。于是，可以归纳地定义$f$的{\bfseries $n$阶导数}，记作
$$f^{(n)}(x), \text{\quad 或 } \dfrac{d^n f(x)}{d x^n}.$$
\pause
约定$f$的$0$阶导数是$f$本身，即$f^{(0)}(x) = f(x).$

\pause

\begin{thm}[Leibniz公式]
设函数$f,g$在区间$I$上都有直到$n$阶导数，那么函数的乘积$f\cdot g$也有$n$阶导数，并且有下述Leibniz公式
$$(f\cdot g)^{(n)} = \sum\limits_{k=0}^n C_n^k f^{(n-k)}g^{(k)}.$$
\end{thm}

\end{frame}

\section{中值定理}

%------------------------------------------------

\begin{frame}

\begin{thm}[拉格朗日中值定理，Lagrange's Mean Value Theorem]<1->
设函数$f$在闭区间$[a,b]$上连续，在开区间$(a,b)$内可微，那么存在点$\xi\in(a,b)$，使得
$$f(b)-f(a) = f'(\xi)(b-a).$$
\end{thm}

\begin{thm}[柯西中值定理，Cauchy's Generalized Mean Value Theorem]<2->
设函数$f,g$在闭区间$[a,b]$上连续，在开区间$(a,b)$内可微，那么存在点$\xi\in(a,b)$，使得
$$g'(\xi)(f(b)-f(a)) = f'(\xi)(g(b)-g(a)).$$
\end{thm}

\pause

中值定理把函数在有限闭区间上的增量跟函数导数联系起来了，因此又叫做有限增量定理（Finite Increment Theorem）。

\end{frame}

\section{泰勒公式}

%------------------------------------------------

\subsection{泰勒多项式与余项}

%------------------------------------------------

\begin{frame}

\begin{itemize}
\item<1-> 两个函数在某点越是有更多的导数（包括$0$阶导数）相同，他们在这点的邻域内就越近似。
\item<2-> 想用尽量``简单''的函数（多项式）去逼近复杂的函数。
\end{itemize}

\pause

\begin{block}{泰勒公式，Taylor's Formula}
设函数$f$在点$x_0$的一个邻域内由定义，且在$x_0$处有$n$阶导数，那么我们有泰勒公式
{\small
$$f(x) = \underbrace{f(x_0) + \dfrac{f'(x_0)}{1!}(x-x_0) + \cdots + \dfrac{f^{(n)}(x_0)}{n!}(x-x_0)^n}_{\text{$f(x)$在$x_0$的$n$阶泰勒多项式}} + \underbrace{r_n(x_0;x)}_{\text{余项}}.$$
}
\end{block}

\pause

Lagrange余项：\quad $r_n(x_0;x) = \dfrac{1}{(n+1)!}f^{(n+1)}(\xi)(x-x_0)^{n+1}$\newline
前提：$f$在$x_0$有$n+1$阶导数

\end{frame}

%------------------------------------------------

\begin{frame}

来看一些初等函数的例子，$x_0=0.$

\pause

\begin{align*}
e^x & = 1 + \dfrac{1}{1!}x + \cdots + \dfrac{1}{n!}x^n + \text{余项} \\
\ln(1+x) & = x - \dfrac{1}{2}x^2 + \dfrac{1}{3}x^3 - \cdots + \dfrac{(-1)^{n-1}}{n}x^n + \text{余项} \\
\sin x & = x - \dfrac{1}{3!}x^3 + \cdots + \dfrac{(-1)^n}{(2n+1)!}x^{2n+1} + \text{余项} \\
\cos x & = 1 - \dfrac{1}{2!}x^2 + \cdots + \dfrac{(-1)^n}{(2n)!}x^{2n} + \text{余项} \\
\end{align*}

\end{frame}

%------------------------------------------------

\subsection{泰勒公式的应用}

%------------------------------------------------

\begin{frame}

\begin{itemize}
\item<1-> 做近似计算：
考虑$\sin x, x\in[-1,1]$，用泰勒多项式逼近它，使得绝对误差不超过$0.001$。\newline
解：考虑Lagrange余项$r_{2n+2}(0;x) = \frac{\sin(\xi+\frac{\pi}{2}(2n+3))}{(2n+3)!}x^{2n+3}$。$x\in[-1,1]$时，它的绝对值$|r_{2n+2}(0;x)| \leqslant \frac{1}{(2n+3)!}$。此时，只要$n=2$即可满足绝对误差不超过$0.001$。因此可以写
$$\sin x \approx x - \dfrac{1}{3!}x^3 + \dfrac{1}{5!}x^5$$
\item<2-> 求不定式（$\frac{0}{0}$型或$\frac{\infty}{\infty}$）的极限：
\begin{align*}
& \lim\limits_{x \to 0} \dfrac{\cos x - e^{-x^2/2}}{x^4} \\
= & \lim\limits_{x \to 0} \dfrac{\left(1 - \frac{x^2}{2!} + \frac{x^4}{4!}\right) - \left(1 + (-\frac{x^2}{2}) + \frac{1}{2!}(-\frac{x^2}{2})^2\right) + \text{\small 更高阶无穷小}}{x^4} \\
= & \lim\limits_{x \to 0} \dfrac{-\frac{1}{12}x^4 + \text{\small 更高阶无穷小}}{x^4} = -\dfrac{1}{12}
\end{align*}
\end{itemize}

\end{frame}

\section{洛必达法则}

%------------------------------------------------

\begin{frame}

\begin{thm}[洛必达法则，L'Hôpital's Rule]
设函数$f,g$在$a$的近旁可微，且$g'(x)\neq0$，且$\lim\limits_{x\to a}\dfrac{f'(x)}{g'(x)}$存在。如果下列2个条件之一成立
\begin{itemize}
\item $\lim\limits_{x\to a} f(x) = \lim\limits_{x\to a} g(x) = 0,$
\item $\lim\limits_{x\to a} f(x) = \lim\limits_{x\to a} g(x) = \infty,$
\end{itemize}
则会有
$$\lim\limits_{x\to a}\dfrac{f(x)}{g(x)} = \lim\limits_{x\to a}\dfrac{f'(x)}{g'(x)}$$
\end{thm}

\end{frame}

%------------------------------------------------

\begin{frame}

例如之前举的例子$\lim\limits_{x \to 0} \dfrac{\cos x - e^{-x^2/2}}{x^4}$就是一个$\frac{0}{0}$型的不定式的极限。应用洛必达法则，有
\begin{align*}
& \lim\limits_{x \to 0} \dfrac{\cos x - e^{-x^2/2}}{x^4} \\
= & \lim\limits_{x \to 0} \dfrac{-\sin x + xe^{-x^2/2}}{4x^3} \\
= & \lim\limits_{x \to 0} \dfrac{-\cos x + (1-x^2)e^{-x^2/2}}{12x^2} \\
= & \lim\limits_{x \to 0} \dfrac{\sin x + (x^3-3x)e^{-x^2/2}}{24x} \\
= & \lim\limits_{x \to 0} \dfrac{\cos x - (x^4-6x^2 + 3)e^{-x^2/2}}{24} \\
= & \dfrac{1-3}{24} = -\dfrac{1}{12}
\end{align*}

\end{frame}

\section{极值与凸性}

%------------------------------------------------

\begin{frame}

在机器学习训练模型的时候，粗略地来说，我们很多时候都是在找损失函数（Loss Function）的期望取得或接近极（最）小值的那组参数。

我们来看看如何用导数来描述极值。

\pause

\begin{block}{极值点的必要条件}
设函数$f$在$x_0$的一个邻域内有定义。要使$x_0$是$f$的一个极值点，那么以下条件之一必须成立：
\begin{itemize}
\item $f$在$x_0$不可微；
\item $f'(x_0) = 0.$
\end{itemize}
\end{block}

\end{frame}
%------------------------------------------------

\begin{frame}

\begin{block}{用一阶导数表达的极值的充分条件}
设函数$f$在$x_0$的一个邻域$U(x_0)$内有定义，在$x_0$处连续，且在去心邻域$\mathring{U}(x_0)$内可微，那么
\begin{itemize}
\item 若$f'(x)$在$\mathring{U}(x_0)$上恒正或恒负，那么$f$在$x_0$处没有极值；
\item 若$f'(x)$在$\mathring{U}^-(x_0)$上恒正（resp. 恒负），且在$\mathring{U}^+(x_0)$上恒负（resp. 恒正），那么$x_0$是$f$的严格局部极大值点（resp. 严格局部极小值点）。
\end{itemize}
\end{block}

\pause

简单来说:
\begin{align*}
\text{导数不变号} & \Longrightarrow \text{无极值} \\
\text{导数变号} & \Longrightarrow \text{有极值} \\
\end{align*}


\end{frame}

\begin{frame}

\begin{block}{用高阶导数表达的极值的充分条件}
设$f$在$x_0$处有$n$阶导数（$n>1$），且$f^{(1)}(x_0) = \cdots = f^{(n-1)}(x_0) = 0, f^{(n)}(x_0) \neq 0,$ 那么
\begin{itemize}
\item 若$n$为奇数，那么$f$在$x_0$处没有极值；
\item 若$n$为偶数，那么当$f^{(n)}(x_0) > 0$时$f$在$x_0$处有严格局部极小值；当$f^{(n)}(x_0) < 0$时$f$在$x_0$处有严格局部极大值。
\end{itemize}
\end{block}

\pause

\vspace{2em}

注意，以上的充分条件都不是必要条件！

\end{frame}

\begin{frame}

很多时候，出于进行计算的时候收敛速度的考虑，我们取的损失函数是一个凸函数。 

\pause

\begin{block}{凸函数定义}
设函数$f$在区间$I$上有定义，若对任意$x_1,x_2\in I,$ 以及任意的$\alpha_1,\alpha_2 \geqslant 0, \alpha_1 + \alpha_2 = 1,$ 都有
$$f(\alpha_1x_1 + \alpha_2x_2) \leqslant \alpha_1f(x_1) + \alpha_2f(x_2),$$
则称$f$是$I$上的{\bfseries 凸函数}。进一步，如果当$x_1 \neq x_2, \alpha_1\cdot\alpha_2 \neq 0$时，上述不等式是严格的（``$\leqslant$''换成``$<$''），则称$f$是{\bfseries 严格凸函数}。
\end{block}

\pause

\begin{frame}

\begin{block}{可微函数凸性判别法}<1->
设函数$f$在开区间$(a,b)$上可微，那么$f$是凸函数（resp. 严格凸函数）当且仅当它的导函数$f'$在$(a,b)$上不减（resp. 严格递增）。
\end{block}

\begin{remark}<2->
如果$f$还有2阶导数，那么上述充要条件可替换为：在$(a,b)$上$f''\geqslant0$（resp. $f''>0$）。
\end{remark}

\pause

\begin{block}{延森不等式，Jensen's Inequality}
设$f$是开区间$(a,b)$上的凸函数，$x_1,\ldots,x_n \in (a,b)$, $\alpha_1,\ldots,\alpha_n \geqslant 0$且满足$\alpha_1+\cdots+\alpha_n = 1,$ 那么有不等式
$$f(\alpha_1x_1+\cdots+\alpha_nx_n) \leqslant \alpha_1f(x_1)+\cdots+\alpha_nf(x_n).$$
\end{block}

\end{frame}
